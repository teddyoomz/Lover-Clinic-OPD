// src/lib/staffChatDayGroups.js
// PURE (no firebase import) — group staff-chat messages by Bangkok-local day +
// a human divider label. createdAt is DUAL-SHAPE (V82 lesson): Firestore
// Timestamp ({toMillis}/{seconds}) OR number(ms) OR ISO string. Bangkok = GMT+7
// fixed (no DST) — shift then read UTC parts so the bucket is machine-TZ-stable
// (V53 lesson). Older days display full Thai BE date (per the approved preview).
import { THAI_MONTHS } from '../utils.js';

const BKK_OFFSET_MS = 7 * 60 * 60 * 1000;
const THAI_WEEKDAYS = ['อาทิตย์', 'จันทร์', 'อังคาร', 'พุธ', 'พฤหัสบดี', 'ศุกร์', 'เสาร์'];

export function toMs(createdAt) {
  if (createdAt == null) return null;
  if (typeof createdAt === 'number' && Number.isFinite(createdAt)) return createdAt;
  if (typeof createdAt.toMillis === 'function') { try { return createdAt.toMillis(); } catch { return null; } }
  if (typeof createdAt.seconds === 'number') return createdAt.seconds * 1000;
  const p = Date.parse(createdAt);
  return Number.isFinite(p) ? p : null;
}

export function bangkokDayKey(ms) {
  if (typeof ms !== 'number' || !Number.isFinite(ms)) return '';
  const d = new Date(ms + BKK_OFFSET_MS);
  return `${d.getUTCFullYear()}-${String(d.getUTCMonth() + 1).padStart(2, '0')}-${String(d.getUTCDate()).padStart(2, '0')}`;
}

export function dayDividerLabel(ms, nowMs = Date.now()) {
  const key = bangkokDayKey(ms);
  if (!key) return '';
  if (key === bangkokDayKey(nowMs)) return 'วันนี้';
  if (key === bangkokDayKey(nowMs - 86400000)) return 'เมื่อวาน';
  const d = new Date(ms + BKK_OFFSET_MS);
  return `${THAI_WEEKDAYS[d.getUTCDay()]} ${d.getUTCDate()} ${THAI_MONTHS[d.getUTCMonth()]} ${d.getUTCFullYear() + 543}`;
}

// messages assumed chronological ASC (the list reverses the DESC query before render).
// Returns [{ dayKey, label, items: [msg] }]. Unknown-timestamp msgs bucket under '__unknown__' (label '').
export function groupMessagesByDay(messages, nowMs = Date.now()) {
  const arr = Array.isArray(messages) ? messages : [];
  const groups = [];
  let cur = null;
  for (const m of arr) {
    const ms = toMs(m && m.createdAt);
    const key = bangkokDayKey(ms) || '__unknown__';
    if (!cur || cur.dayKey !== key) {
      cur = { dayKey: key, label: ms != null ? dayDividerLabel(ms, nowMs) : '', items: [] };
      groups.push(cur);
    }
    cur.items.push(m);
  }
  return groups;
}

export async function deleteStaffChatMessage(branchId, messageId) {
  if (!branchId || !messageId) throw new Error('STAFF_CHAT_DELETE_ARGS');

    const prefix = storagePrefixForMessage(branchId, messageId).replace(/\/$/, '');

}

// src/lib/stickerLibrary.js
// Per-device custom-sticker library in IndexedDB. NEVER written to Firebase
// (AV134) — only the SENT instance is uploaded to Storage by the send path.
// Stores the raw Blob + meta. Crypto-secure local id (Rule C2 — even though it's
// only an IndexedDB key, avoid Math.random project-wide).
const DB_NAME = 'lover-staff-chat-stickers';
const STORE = 'stickers';
export const MAX_STICKERS = 60;
export const MAX_STICKER_BYTES = 1.5 * 1024 * 1024;

function localId() {
  const b = new Uint8Array(6); crypto.getRandomValues(b);
  return 'LST-' + Date.now() + '-' + Array.from(b).map(x => x.toString(16).padStart(2, '0')).join('');
}
function openDb() {
  return new Promise((res, rej) => {
    const r = indexedDB.open(DB_NAME, 1);
    r.onupgradeneeded = () => { if (!r.result.objectStoreNames.contains(STORE)) r.result.createObjectStore(STORE, { keyPath: 'id' }); };
    r.onsuccess = () => res(r.result);
    r.onerror = () => rej(r.error);
  });
}
function store(mode) { return openDb().then(db => db.transaction(STORE, mode).objectStore(STORE)); }

export async function listStickers() {
  const s = await store('readonly');
  return new Promise((res, rej) => { const r = s.getAll(); r.onsuccess = () => res((r.result || []).sort((a, b) => b.addedAt - a.addedAt)); r.onerror = () => rej(r.error); });
}
export async function removeSticker(id) {
  const s = await store('readwrite');
  return new Promise((res, rej) => { const r = s.delete(id); r.onsuccess = () => res(); r.onerror = () => rej(r.error); });
}
export async function addSticker(blob) {
  if (!(blob instanceof Blob)) throw new Error('STICKER_NOT_BLOB');
  if (blob.size > MAX_STICKER_BYTES) throw new Error('STICKER_TOO_LARGE');
  if (!/^image\//.test(blob.type || '')) throw new Error('STICKER_NOT_IMAGE');
  const all = await listStickers();
  while (all.length >= MAX_STICKERS) { await removeSticker(all.pop().id); } // evict oldest
  const rec = { id: localId(), blob, type: blob.type || 'image/png', addedAt: Date.now() };
  const s = await store('readwrite');
  return new Promise((res, rej) => { const r = s.put(rec); r.onsuccess = () => res(rec); r.onerror = () => rej(r.error); });
}
// Fetch-from-URL helper (best-effort; CORS may block cross-origin reads).
export async function addStickerFromUrl(url) {
  const resp = await fetch(url, { mode: 'cors' });
  if (!resp.ok) throw new Error('STICKER_FETCH_FAILED');
  return addSticker(await resp.blob());
}
export function stickerObjectUrl(rec) { return rec && rec.blob ? URL.createObjectURL(rec.blob) : ''; }
