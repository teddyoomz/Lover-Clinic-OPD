// scripts/perf-lib.mjs — shared perf-measurement library (Playwright library mode + CDP)
// Pure helpers exported for vitest; browser drivers used by perf-baseline/visual-parity.
export const median = (a) => {
  const s = [...a].sort((x, y) => x - y); const m = Math.floor(s.length / 2);
  return s.length % 2 ? s[m] : (s[m - 1] + s[m]) / 2;
};
export const aggregateRuns = (runs) => {
  const out = {};
  for (const k of Object.keys(runs[0] || {})) out[k] = median(runs.map(r => r[k] ?? 0));
  return out;
};

export async function injectStaffAuth(context, baseURL) {
  // Mirror tests/e2e/helpers.js: REST signInWithPassword → localStorage injection.
  const apiKey = process.env.VITE_FIREBASE_API_KEY || process.env.FIREBASE_API_KEY;
  const email = process.env.E2E_STAFF_EMAIL, password = process.env.E2E_STAFF_PASSWORD;
  if (!apiKey || !email || !password) throw new Error('need FIREBASE_API_KEY + E2E_STAFF_EMAIL/PASSWORD in env (.env.local.prod pulled per Rule R)');
  const res = await fetch(`https://identitytoolkit.googleapis.com/v1/accounts:signInWithPassword?key=${apiKey}`, {
    method: 'POST', headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ email, password, returnSecureToken: true }),
  });
  const tok = await res.json();
  if (!tok.idToken) throw new Error('staff sign-in failed: ' + JSON.stringify(tok).slice(0, 200));
  // (copy the EXACT authUser object shape from tests/e2e/helpers.js — single source; do not improvise)
  await context.addInitScript(([key, user]) => {
    localStorage.setItem(`firebase:authUser:${key}:[DEFAULT]`, JSON.stringify(user));
  }, [apiKey, buildAuthUser(tok, email)]);
}

export function metricsInitScript() {
  // Runs before page scripts: observers buffered from t=0.
  return `(() => {
    window.__perf = { longTasks: [], lcp: 0, fcp: 0, cls: 0, idleMutations: 0 };
    try { new PerformanceObserver(l => l.getEntries().forEach(e => __perf.longTasks.push(e.duration))).observe({ type: 'longtask', buffered: true }); } catch {}
    try { new PerformanceObserver(l => { const es = l.getEntries(); if (es.length) __perf.lcp = es[es.length-1].startTime; }).observe({ type: 'largest-contentful-paint', buffered: true }); } catch {}
    try { new PerformanceObserver(l => l.getEntries().forEach(e => { if (e.name === 'first-contentful-paint') __perf.fcp = e.startTime; })).observe({ type: 'paint', buffered: true }); } catch {}
    try { new PerformanceObserver(l => l.getEntries().forEach(e => { if (!e.hadRecentInput) __perf.cls += e.value; })).observe({ type: 'layout-shift', buffered: true }); } catch {}
  })();`;
}

export async function collectMetrics(page) {
  // 5s idle window for mutation counting (re-render-storm proxy)
  await page.evaluate(() => new Promise(res => {
    const mo = new MutationObserver(ms => { window.__perf.idleMutations += ms.length; });
    mo.observe(document.body, { subtree: true, childList: true, attributes: true, characterData: true });
    setTimeout(() => { mo.disconnect(); res(); }, 5000);
  }));
  return page.evaluate(() => {
    const p = window.__perf;
    const js = performance.getEntriesByType('resource').filter(r => r.name.match(/\\.js(\\?|$)/));
    return {
      FCP_ms: Math.round(p.fcp), LCP_ms: Math.round(p.lcp), CLS: +p.cls.toFixed(4),
      longTasks_count: p.longTasks.length,
      longTasks_totalMs: Math.round(p.longTasks.reduce((a, b) => a + b, 0)),
      jsTransferred_KB: Math.round(js.reduce((a, r) => a + (r.transferSize || 0), 0) / 1024),
      jsResources: js.length,
      domNodes: document.querySelectorAll('*').length,
      heapUsed_MB: performance.memory ? Math.round(performance.memory.usedJSHeapSize / 1048576) : 0,
      idleMutations_5s: p.idleMutations,
    };
  });
}

// local-preview: expects `npm run build && npx vite preview --port 4173` already running (low-noise primary)
// prod: --target prod → https://lover-clinic-app.vercel.app (secondary, network-real)
import { chromium } from '@playwright/test';
import { mkdirSync, writeFileSync, readFileSync, existsSync } from 'fs';
import { SURFACES, injectStaffAuth, metricsInitScript, collectMetrics, aggregateRuns } from './perf-lib.mjs';

const arg = (k, d) => { const i = process.argv.indexOf('--' + k); return i > -1 ? process.argv[i + 1] : d; };
const runId = arg('run', 'baseline'), target = arg('target', 'local-preview'), only = arg('surface', null);
const BASE = target === 'prod' ? 'https://lover-clinic-app.vercel.app' : 'http://localhost:4173';
const links = existsSync('docs/perf/links.json') ? JSON.parse(readFileSync('docs/perf/links.json', 'utf8')) : {};
const RUNS = 3;

const browser = await chromium.launch();
const outDir = `docs/perf/${runId}-${target}`; mkdirSync(outDir, { recursive: true });
for (const s of SURFACES.filter(x => !only || x.id === only)) {
  const url = BASE + s.url.replace('{SCHEDULE_TOKEN}', links.schedule || '')
                         .replace('{PATIENT_TOKEN}', links.patient || '')
                         .replace('{SESSION_ID}', links.session || '');
  if (url.includes('{')) { console.log(`skip ${s.id} (no link token)`); continue; }
  const runs = [];
  for (let i = 0; i < RUNS; i++) {
    const ctx = await browser.newContext({ viewport: { width: 1280, height: 800 } });
    if (s.auth) await injectStaffAuth(ctx, BASE);
    const page = await ctx.newPage();
    await page.addInitScript(metricsInitScript());
    await page.goto(url, { waitUntil: 'networkidle', timeout: 60000 }).catch(() => {});
    const m = await collectMetrics(page);
    if (s.interaction) {
      const t0 = Date.now();
      await page.click(s.interaction.clickSel).catch(() => {});
      await page.waitForLoadState('networkidle').catch(() => {});
      m.interaction_ms = Date.now() - t0;
    }
    runs.push(m); await ctx.close();
  }
  const rec = { runId, target, surface: s.id, url: s.url, runs: RUNS, metrics: aggregateRuns(runs) };
  writeFileSync(`${outDir}/${s.id}.json`, JSON.stringify(rec, null, 2));
  console.log(s.id, JSON.stringify(rec.metrics));
}
await browser.close();

import { diffRatio } from '../scripts/perf-visual-parity.mjs';
it('diffRatio: identical buffers → 0; one-channel jump > tol → counted', () => {
  const w = 2, h = 1, ch = 3;
  const a = Buffer.from([10, 10, 10, 200, 200, 200]);
  const b = Buffer.from([10, 10, 10, 200, 200, 200]);
  expect(diffRatio(a, b, w, h, ch, 24)).toBe(0);
  const c = Buffer.from([10, 10, 10, 100, 200, 200]);   // pixel 2 red Δ=100 > 24
  expect(diffRatio(a, c, w, h, ch, 24)).toBe(0.5);
});
// scripts/perf-visual-parity.mjs
// --capture <setName> : screenshot every SURFACE × {dark,light} → docs/perf/shots/<setName>/
// --diff <setA> <setB> : sharp raw-pixel diff per pair → PASS/FAIL table (threshold 0.5%)
// Theme switch: localStorage 'app-theme' via addInitScript. Animations frozen:
//   page.emulateMedia({ reducedMotion: 'reduce' }) + CSS injection `*{animation:none!important;transition:none!important;caret-color:transparent}`
// Dynamic data caveat: same local preview + same Firestore data, minutes apart → real drift small;
//   any surface > threshold → emit side-by-side for HUMAN eyeball (Rule Q-vis: screenshot wins).
export function diffRatio(a, b, w, h, channels, tol) {
  let diff = 0;
  for (let p = 0; p < w * h; p++) {
    const i = p * channels;
    if (Math.abs(a[i]-b[i]) > tol || Math.abs(a[i+1]-b[i+1]) > tol || Math.abs(a[i+2]-b[i+2]) > tol) diff++;
  }
  return diff / (w * h);
}
