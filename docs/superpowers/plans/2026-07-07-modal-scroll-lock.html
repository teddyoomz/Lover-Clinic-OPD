// src/lib/useModalScrollLock.js  (NEW — sibling ของ useEscToClose.js)
import { useEffect } from 'react';

// AV205 — module-level ref count: modal ซ้อน modal ปลดล็อคเมื่อตัวสุดท้ายปิดเท่านั้น
let lockCount = 0;

function acquire() {
  lockCount += 1;
  if (lockCount !== 1) return;
  const el = document.documentElement;
  const gutter = window.innerWidth - el.clientWidth; // กัน layout shift ตอน scrollbar หาย
  if (gutter > 0) el.style.setProperty('--scroll-lock-gutter', `${gutter}px`);
  el.setAttribute('data-modal-open', '1');
}

function release() {
  lockCount = Math.max(0, lockCount - 1);
  if (lockCount !== 0) return;
  const el = document.documentElement;
  el.removeAttribute('data-modal-open');
  el.style.removeProperty('--scroll-lock-gutter');
}

export function useModalScrollLock(active = true) {
  useEffect(() => {
    if (!active || typeof document === 'undefined') return undefined;
    acquire();
    return release;
  }, [active]);
}

// สำหรับ modal ที่เขียน inline ใน host ใหญ่ (AdminDashboard/TFP/SaleTab ฯลฯ) —
// วางเป็น child แรกของ overlay: mount = lock, unmount = unlock (เลี่ยง hook-in-conditional)
export function ModalScrollLock({ active = true }) {
  useModalScrollLock(active);
  return null;
}

// test-only
export function _getLockCount() { return lockCount; }

// helper: scrollTop ของ background scroller หลัก (AdminDashboard/Backend ใช้ inner scroller)
async function bgScrollTops(page) {
  return page.evaluate(() =>
    [...document.querySelectorAll('div')].filter(d => d.scrollHeight > d.clientHeight + 4 &&
      /(auto|scroll)/.test(getComputedStyle(d).overflowY) && !d.closest('[data-modal-open], .fixed'))
      .slice(0, 5).map(d => d.scrollTop)
  );
}
