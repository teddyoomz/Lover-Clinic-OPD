import { render, screen, fireEvent, within } from '@testing-library/react';
import { vi, describe, it, expect, beforeEach } from 'vitest';
import EDScoreBox from '../src/components/backend/EDScoreBox.jsx';

const mockDelete = vi.fn(() => Promise.resolve());
vi.mock('../src/lib/scopedDataLayer.js', () => ({ deleteAssessmentRound: (...a) => mockDelete(...a) }));

// intake round = ADAM(positive)+IIEF, followup round = ADAM+IIEF. (matches the user screenshot)
const intakePerf = { assessmentDate: '2026-05-13', adam_q1: true, iief_q1: 2, iief_q2: 2, iief_q3: 2, iief_q4: 2, iief_q5: 2 };
const assessments = [{ id: 'A2', status: 'completed', assessmentDate: '2026-06-18',
  types: ['adam','iief'], rawAnswers: { adam_q1: true, iief_q1: 3, iief_q2: 3, iief_q3: 3, iief_q4: 2, iief_q5: 2 } }];

const setup = (props = {}) => render(<EDScoreBox customerId="C1" intakePerf={intakePerf} assessments={assessments} isDark={true} onSend={vi.fn()} {...props} />);

describe('EDScoreBox round-select', () => {
  beforeEach(() => mockDelete.mockClear());

  it('R1 default view: header "ครั้งล่าสุด", no back pill', () => {
    setup();
    expect(screen.getByText(/ครั้งล่าสุด:/)).toBeInTheDocument();
    expect(screen.queryByTestId('ed-back-to-latest')).toBeNull();
  });

  it('R2 click round-1 row → header "กำลังดู: ครั้งที่ 1" + back pill appears', () => {
    setup();
    fireEvent.click(screen.getByTestId('ed-history-toggle'));
    fireEvent.click(screen.getByTestId('ed-history-__intake__'));
    expect(screen.getByText(/กำลังดู:/)).toBeInTheDocument();
    expect(screen.getByText(/ครั้งที่ 1/)).toBeInTheDocument();
    expect(screen.getByTestId('ed-back-to-latest')).toBeInTheDocument();
  });

  it('R3 selected round-1 chips show that round only (ADAM present, MRS/PE dash)', () => {
    setup();
    fireEvent.click(screen.getByTestId('ed-history-toggle'));
    fireEvent.click(screen.getByTestId('ed-history-__intake__'));
    const mrs = screen.getByTestId('ed-chip-mrs');
    expect(within(mrs).getByText('—')).toBeInTheDocument();
    const adam = screen.getByTestId('ed-chip-adam');
    expect(within(adam).queryByText(/\(ครั้งที่/)).toBeNull(); // no olderTag in single-round view
  });

  it('R4 back pill → returns to default (header ครั้งล่าสุด, pill gone)', () => {
    setup();
    fireEvent.click(screen.getByTestId('ed-history-toggle'));
    fireEvent.click(screen.getByTestId('ed-history-__intake__'));
    fireEvent.click(screen.getByTestId('ed-back-to-latest'));
    expect(screen.getByText(/ครั้งล่าสุด:/)).toBeInTheDocument();
    expect(screen.queryByTestId('ed-back-to-latest')).toBeNull();
  });

  it('R5 delete icon does NOT select the row (stopPropagation)', () => {
    window.confirm = vi.fn(() => true);
    setup();
    fireEvent.click(screen.getByTestId('ed-history-toggle'));
    fireEvent.click(screen.getByTestId('ed-delete-A2'));
    expect(mockDelete).toHaveBeenCalledWith('A2');
    expect(screen.queryByText(/กำลังดู:/)).toBeNull(); // delete didn't switch to "viewing"
  });

  it('R6 selecting then deleting that round resets to default', async () => {
    window.confirm = vi.fn(() => true);
    setup();
    fireEvent.click(screen.getByTestId('ed-history-toggle'));
    fireEvent.click(screen.getByTestId('ed-history-A2')); // select followup
    expect(screen.getByText(/กำลังดู:/)).toBeInTheDocument();
    fireEvent.click(screen.getByTestId('ed-delete-A2'));
    await screen.findByText(/ครั้งล่าสุด:/); // back to default after delete resolves
    expect(screen.queryByTestId('ed-back-to-latest')).toBeNull();
  });

  it('R7 row is a real button-like control (aria-pressed reflects selection + keyboard)', () => {
    setup();
    fireEvent.click(screen.getByTestId('ed-history-toggle'));
    const row = screen.getByTestId('ed-history-__intake__');
    expect(row).toHaveAttribute('aria-pressed', 'false');
    fireEvent.keyDown(row, { key: 'Enter' });
    expect(screen.getByTestId('ed-history-__intake__')).toHaveAttribute('aria-pressed', 'true');
  });

  it('R8 intake row (รับเข้า, not deletable) is still selectable', () => {
    setup();
    fireEvent.click(screen.getByTestId('ed-history-toggle'));
    fireEvent.click(screen.getByTestId('ed-history-__intake__'));
    expect(screen.getByText(/· รับเข้า/)).toBeInTheDocument();
  });
});
