Heavy reuse of calculateADAM/IIEF/MRS, generateQrDataUrl, patient-link pattern, be_course_changes blueprint.

// src/lib/assessmentRoundsCore.js — PURE. No Firestore, no React.
// Round number is DERIVED (date-rank), never a stored field → delete renumbers.
export const ED_TYPES = ['adam', 'iief', 'mrs', 'pe'];

const perfKeys = (raw) => Object.keys(raw || {}).filter(
  k => /^(adam_|iief_|mrs_)/.test(k) || k === 'symp_pe');
const hasPerf = (raw) => perfKeys(raw).some(k => {
  const v = raw[k]; return v === true || (typeof v === 'string' && v !== '') || (typeof v === 'number' && Number.isFinite(v));
});
const typesInRaw = (raw) => {
  const t = [];
  if (Object.keys(raw||{}).some(k => k.startsWith('adam_') && raw[k])) t.push('adam');
  if (Object.keys(raw||{}).some(k => k.startsWith('iief_') && raw[k] !== '' && raw[k] != null)) t.push('iief');
  if (Object.keys(raw||{}).some(k => k.startsWith('mrs_') && raw[k] !== '' && raw[k] != null)) t.push('mrs');
  if (raw && raw.symp_pe) t.push('pe');
  return t;
};

// intakePerf: pickKioskAssessmentFields(customer.patientData) — {} if none.
// beAssessments: array of be_assessments docs (only status==='completed' counts as a round).
export function deriveRounds(intakePerf, beAssessments) {
  const list = [];
  if (intakePerf && hasPerf(intakePerf)) {
    list.push({ id: '__intake__', source: 'intake', deletable: false,
      assessmentDate: intakePerf.assessmentDate || '', raw: intakePerf,
      types: (intakePerf.__types__ || typesInRaw(intakePerf)) });
  }
  for (const a of (beAssessments || [])) {
    if (a.status && a.status !== 'completed') continue;     // pending links not yet filled → not a round
    list.push({ id: a.id, source: 'followup', deletable: true,
      assessmentDate: a.assessmentDate || '', raw: a.rawAnswers || {},
      types: a.types && a.types.length ? a.types : typesInRaw(a.rawAnswers || {}),
      scores: a.scores, createdBy: a.createdBy });
  }
  // sort by date asc; blank dates last-stable; then assign derived round#
  list.sort((x, y) => String(x.assessmentDate).localeCompare(String(y.assessmentDate)));
  return list.map((r, i) => ({ ...r, round: i + 1 }));
}

export function nextRoundNumber(intakePerf, beAssessments) {
  return deriveRounds(intakePerf, beAssessments).length + 1;
}

// each type → the most-recent round (by date) that measured it, or null.
export function latestPerType(intakePerf, beAssessments) {
  const rounds = deriveRounds(intakePerf, beAssessments);
  const out = { adam: null, iief: null, mrs: null, pe: null };
  for (const r of rounds) for (const t of r.types) out[t] = r; // later (newer) overwrites
  return out;
}

// latest N rounds (newest first) for compact displays (TFP latest-2).
export function latestRounds(intakePerf, beAssessments, n = 2) {
  const rounds = deriveRounds(intakePerf, beAssessments);
  return rounds.slice(-n).reverse();
}

export function scoreForType(type, raw) {

// Removes the auto-generated "ผลการคัดกรองอาการ / Clinical Screening Results" block
// (the generateClinicalSummary section, utils.js:407-422 TH / :469-483 EN) from a stored note.
// Boundary = the screening header line → up to the next separator line / section / EOF.
const SCREEN_HEADERS = ['ผลการคัดกรองอาการ', 'Clinical Screening Results'];
const SEP_RE = /^[\s─=_-]{4,}\s*$/;
export function stripScreeningSection(note) {
  if (!note || typeof note !== 'string') return note || '';
  const lines = note.split(/\r?\n/);
  const out = []; let skipping = false;
  for (let i = 0; i < lines.length; i++) {
    const ln = lines[i];
    if (!skipping && SCREEN_HEADERS.some(h => ln.trim().startsWith(h))) {
      skipping = true;
      // also drop a separator immediately preceding the header we already pushed
      if (out.length && SEP_RE.test(out[out.length-1])) out.pop();
      continue;
    }
    if (skipping) {
      // stop skipping at the next separator OR a non-indented new section header
      if (SEP_RE.test(ln)) { skipping = false; continue; } // consume the closing sep
      if (ln.trim() === '' && i+1 < lines.length && !/^\s/.test(lines[i+1] || '')) { skipping = false; out.push(ln); }
      continue;
    }
    out.push(ln);
  }
  return out.join('\n').replace(/\n{3,}/g, '\n\n').trim();
}

export function listenToAssessments(customerId, onChange, onError) {
  if (!customerId) { onChange?.([]); return () => {}; }
  const q = query(assessmentsCol(), where('customerId', '==', String(customerId)));
  return onSnapshot(q, (snap) => {

    items.sort((a,b) => String(a.assessmentDate||'').localeCompare(String(b.assessmentDate||'')));
    onChange?.(items);
  }, (e) => onError?.(e));
}
