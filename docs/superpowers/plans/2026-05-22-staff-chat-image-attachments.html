T1 · Foundation: retention core + image lib

Create staffChatRetentionCore.js (pure); extend staffChatImageResize.js.

// staffChatRetentionCore.js
export const RETENTION_DAYS=30, ORPHAN_GRACE_MS=86400000;
export const storagePrefixForMessage=(b,m)=>`staff-chat-attachments/${b}/${m}/`;
export const isExpired=(ms,now,d=RETENTION_DAYS)=>Number.isFinite(ms)&&(now-ms)>d*86400000;
export function extractStoragePathFromUrl(u){if(typeof u!=='string')return null;const m=u.match(/\/o\/([^?]+)/);if(!m)return null;try{return decodeURIComponent(m[1]);}catch{return null;}}
export const isOrphanFolder=({docExists,folderMs,now,grace=ORPHAN_GRACE_MS})=>!docExists&&(typeof folderMs!=='number'||(now-folderMs)>grace);
export function gridLayoutFor(n){if(n<=1)return{cls:'sc-g1',show:1,overflow:0};if(n===2)return{cls:'sc-g2',show:2,overflow:0};if(n===3)return{cls:'sc-g3',show:3,overflow:0};return{cls:'sc-g4',show:4,overflow:n-4};}
// staffChatImageResize.js (add)
export const STAFF_CHAT_MAX_IMAGES=10, STAFF_CHAT_MAX_BYTES=52428800;
export const STAFF_CHAT_ALLOWED=['image/jpeg','image/png','image/webp','image/gif'], STAFF_CHAT_THUMB=512;
export function validateStaffChatImage(f){if(!f||!STAFF_CHAT_ALLOWED.includes(f.type))return{ok:false,message:'รองรับเฉพาะ JPEG/PNG/WebP/GIF (HEIC: แปลงเป็น JPEG ก่อน)'};if(f.size>STAFF_CHAT_MAX_BYTES)return{ok:false,message:'ไฟล์ใหญ่เกิน 50MB'};return{ok:true};}
export const extForMime=m=>({'image/jpeg':'jpg','image/png':'png','image/webp':'webp','image/gif':'gif'}[m]||'jpg');
export function staffChatImagePaths(b,m,id,ext){const base=`staff-chat-attachments/${b}/${m}/${id}`;return{thumbPath:`${base}-t.jpg`,fullPath:`${base}-o.${ext}`};}
