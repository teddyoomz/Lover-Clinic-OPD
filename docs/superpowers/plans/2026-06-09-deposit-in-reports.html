const depositsColReports = () => collection(db, ...basePath(), 'be_deposits');
export async function loadDepositsByDateRange({ from='', to='', branchId='', allBranches=false } = {}) {
  const wantBranch = shouldFilterByBranch({ branchId, allBranches });
  try {
    const conds = [];
    if (from) conds.push(where('paymentDate','>=',from));
    if (to)   conds.push(where('paymentDate','<=',to));
    if (wantBranch) conds.push(where('branchId','==',branchId));
    const q = conds.length ? query(depositsColReports(), ...conds, orderBy('paymentDate','desc'))
                           : query(depositsColReports(), orderBy('paymentDate','desc'));
    const snap = await getDocs(q);
    return snap.docs.map(d => ({ ...d.data(), id: d.id }));
  } catch {
    const snap = await getDocs(depositsColReports());
    let items = snap.docs.map(d => ({ ...d.data(), id: d.id }));
    if (from) items = items.filter(x => (x.paymentDate||'') >= from);
    if (to)   items = items.filter(x => (x.paymentDate||'') <= to);
    if (wantBranch) items = items.filter(x => x.branchId === branchId);
    items.sort((a,b)=>(b.paymentDate||'').localeCompare(a.paymentDate||''));
    return items;
  }
}

import { roundTHB } from './reportsUtils.js';
// status that means money genuinely came in (cancelled = reversed/never received)
const RECEIVED_STATUSES = new Set(['active','partial','used','refunded','expired']);
export function depositsReceivedInRange(deposits, { from='', to='' } = {}) {
  return (Array.isArray(deposits)?deposits:[]).filter(d => {
    if (!d || d.status === 'cancelled') return false;
    const pd = String(d.paymentDate || '');
    if (from && pd < from) return false;
    if (to && pd > to) return false;
    return true;
  });
}
export function sumSystemRemainingDeposits(deposits) {
  let t = 0;
  for (const d of (Array.isArray(deposits)?deposits:[])) {
    if (d && (d.status==='active'||d.status==='partial')) t += Number(d.remainingAmount)||0;
  }
  return roundTHB(t);
}
export function buildDepositDeepLinkUrl(depositId) {
  const origin = (typeof window!=='undefined' && window.location?.origin) || '';
  return `${origin}?backend=1&tab=finance&subtab=deposit&deposit=${encodeURIComponent(String(depositId||''))}`;
}
