/**
 * @param {object|null|undefined} session - opd_sessions doc shape
 * @returns {boolean} true iff session has been saved to be_customers via handleOpdClick
 */
export function isOpdSessionSaved(session) {
  if (!session || typeof session !== 'object') return false;
  return !!(session.opdRecordedAt && session.brokerStatus === 'done');
}

/**
 * @param {object|null|undefined} session
 * @returns {boolean} true iff customer has filled the PatientForm (patientData non-empty)
 */
export function hasPatientData(session) {
  if (!session || typeof session !== 'object') return false;
  const pd = session.patientData;
  if (!pd || typeof pd !== 'object') return false;
  return Object.keys(pd).length > 0;
}

/**
 * Resolves the card-level OPD lifecycle state for a given appointment.
 * Returns one of 'A' | 'B' | 'C' | 'D' | 'E' per the V118 state machine.
 *
 * @param {object} args
 * @param {object} args.appt - be_appointments doc shape
 * @param {object|null} args.linkedSession - opd_sessions doc (or null if none/not yet loaded)
 * @returns {'A'|'B'|'C'|'D'|'E'}
 */
export function resolveCardOpdState({ appt, linkedSession }) {
  if (!appt) return 'B';
  // State A: customer already linked (has be_customers record)
  if (appt.customerId) return 'A';
  // No customer linked → check session presence + state
  if (!appt.linkedOpdSessionId) return 'B';
  // Has linkedOpdSessionId but session not yet resolved (lazy fetch pending) — treat as C
  if (!linkedSession) return 'C';
  // Session exists
  if (isOpdSessionSaved(linkedSession)) return 'E';
  if (!hasPatientData(linkedSession)) return 'C';
  return 'D';
}

/**
 * Synthesize a session-shaped object from a be_customers doc for the
 * "ดูข้อมูล" modal when no real opd_sessions doc exists (State A path).
 * Carries `__synthetic: true` so the modal can gate destructive buttons.
 *
 * @param {object} customer - be_customers doc
 * @param {object} appt - the originating be_appointments doc (for context)
 * @returns {object} synth session
 */
export function synthesizeSessionFromCustomer(customer, appt) {
  if (!customer) return null;
  return {
    id: `synth-${customer.id || customer.proClinicId || 'unknown'}-${appt?.id || 'noappt'}`,
    patientData: customer.patientData || {},
    opdRecordedAt: customer.createdAt || null,
    brokerStatus: 'done',
    brokerProClinicId: customer.proClinicId || null,
    brokerProClinicHN: customer.proClinicHN || customer.HN || '',
    customerId: customer.id || customer.proClinicId || '',
    __synthetic: true,
    __synthSourceCustomerId: customer.id || customer.proClinicId || '',
  };
}

// V118 (2026-05-23) — resolve the opd_sessions doc for a given appt.
// Three sources (in order): memo'd current-window sessions → lazy-fetch cache →
// trigger lazy fetch (returns null this render, re-renders on resolve).
//
// NOTE: lazyFetchedTick is read in the dep array so React re-runs this when a
// lazy fetch resolves. The `_` suffix on the variable means "intentionally
// unused-as-reference; needed for re-render trigger".
const resolveLinkedSession = useCallback((appt) => {
  if (!appt?.linkedOpdSessionId) return null;
  const id = appt.linkedOpdSessionId;
  if (sessionsById.has(id)) return sessionsById.get(id);
  if (lazyFetchedSessionsRef.current.has(id)) return lazyFetchedSessionsRef.current.get(id);
  // Trigger lazy fetch (coalesce duplicates via in-flight Set).
  if (!lazyFetchInFlightRef.current.has(id)) {
    lazyFetchInFlightRef.current.add(id);
    // opd_sessions live at root collection 'opd_sessions/{id}' per existing
    // patterns elsewhere in AdminDashboard (search for an existing
    // doc(db, 'opd_sessions', ...) usage to confirm path).
    const ref = doc(db, 'artifacts', appId, 'public', 'data', 'opd_sessions', id);
    getDoc(ref).then(snap => {
      if (snap.exists()) {
        lazyFetchedSessionsRef.current.set(id, { id: snap.id, ...snap.data() });
        setLazyFetchedTick(t => t + 1);
      }
    }).catch(e => {
      console.warn('[V118 resolveLinkedSession] lazy fetch failed:', e);
    }).finally(() => {
      lazyFetchInFlightRef.current.delete(id);
    });
  }
  return null;
  // eslint-disable-next-line react-hooks/exhaustive-deps
}, [sessionsById, lazyFetchedTick, appId, db]);

### AV118 — Card-level OPD state derivation MUST use opdSessionState helpers (2026-05-23 V118 card OPD lifecycle row)

**Class-of-bug**: V12 multi-reader-sweep at the OPD-save-state predicate boundary. Pre-V118, the predicate `(session.opdRecordedAt && session.brokerStatus === 'done')` was inlined at ≥3 sites (AdminDashboard.jsx:3475, :5747; V87/AV84 test). When the state model evolves (e.g. a new "partial-save" intermediate state), inline callsites drift silently.

**Rule**: every derivation of "OPD session saved" / "customer filled" / "card OPD state" in `src/components/admin/**` + `src/pages/AdminDashboard.jsx` MUST go through `src/lib/opdSessionState.js` helpers:
- `isOpdSessionSaved(session)` — replaces inline `session.opdRecordedAt && session.brokerStatus === 'done'`
- `hasPatientData(session)` — replaces inline `session.patientData && Object.keys(...).length`
- `resolveCardOpdState({appt, linkedSession})` — single source for the 5-state machine
- `synthesizeSessionFromCustomer(customer, appt)` — single source for synth-session shape

**Sanctioned exceptions** (closed list):
- `src/lib/opdSessionState.js` itself — defines the helpers, contains the literal `'done'` string
- `tests/v118-opd-session-state-helpers.test.js` — tests the helpers, asserts the predicate
- `tests/v87-link-button-opd-save-guard.test.js` — V87 source-grep regression that locks the V87 patient-link guard literal text (different concern: AV84 trigger-site closed list)
- `src/pages/AdminDashboard.jsx:3475` — `handleOpdClick` early-return literal predicate (preserved per V87/AV84). MUST also call `isOpdSessionSaved` (defense-in-depth) but the literal stays for V87 test compatibility.

**Source-grep regression test**: `tests/v118-card-opd-lifecycle-row-source-grep.test.js` AV118.G1-G4.

**Origin**: V118 (2026-05-23) — card-level OPD lifecycle row introduced 4 new callsites of the saved-predicate. Centralizing prevents future drift.
