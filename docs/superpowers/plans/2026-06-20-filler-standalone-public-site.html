For agentic workers: REQUIRED SUB-SKILL: Use superpowers:executing-plans (inline — recommended for this repo; subagents thrash on the large baseline per V81/V86 and the filler files are tightly coupled). Steps use HTML checkboxes. Spec: docs/superpowers/specs/2026-06-20-filler-standalone-public-site-design.html.

Goal: Serve the existing ?play=filler simulator from a separate Vercel project at loverclinic.vercel.app whose bundle contains ZERO firebase/OPD code — single source, two build targets, one deploy command.

Architecture: Same repo. NEW thin entry (filler.html → src/filler-main.jsx → <FillerSimulator/>) + a separate vite config (vite.filler.config.js → dist-filler/) that tree-shakes away all OPD/firebase code. A build-bundle verifier is the Rule Q L2 proof of the security boundary. og:image generated from the REAL logo via Playwright (already a devDep). Deploy both projects with one npm run deploy:filler.

Tech Stack: Vite 8 · React 19 · vite-plugin-javascript-obfuscator (existing, same scope) · Playwright (existing devDep, for og:image) · Vercel CLI (2 projects via --local-config + VERCEL_PROJECT_ID override).

import { defineConfig } from 'vite'
import react from '@vitejs/plugin-react'
import obfuscator from 'vite-plugin-javascript-obfuscator'

// Standalone PUBLIC build of the filler simulator → dist-filler/.
// Entry = filler.html → src/filler-main.jsx → <FillerSimulator/> ONLY.
// SAME obfuscator scope as vite.config.js (the 4 filler files). publicDir =
// public-filler (only the assets the public page needs — no OPD chart/stickers).
export default defineConfig(({ command }) => ({
  plugins: [
    react(),
    ...(command === 'build' ? [obfuscator({
      include: ['**/fillerMath.js', '**/FillerSimulator.jsx', '**/FillerGraphic2D.jsx', '**/Filler3D.jsx'],
      exclude: ['node_modules/**', 'tests/**'],
      options: {
        compact: true, identifierNamesGenerator: 'hexadecimal', numbersToExpressions: true,
        simplify: true, transformObjectKeys: true, controlFlowFlattening: true,
        controlFlowFlatteningThreshold: 0.75, stringArray: true, stringArrayThreshold: 1,
        stringArrayEncoding: ['base64'], stringArrayCallsTransform: true, splitStrings: true,
        splitStringsChunkLength: 6,
      },
    })] : []),
  ],
  publicDir: 'public-filler',
  build: {
    outDir: 'dist-filler',
    emptyOutDir: true,
    rollupOptions: { input: 'filler.html' },
  },
}))

Task 3: Build-bundle security verifier (Rule Q L2 proof)

Files: Create scripts/verify-filler-bundle.mjs

// Rule Q L2 proof for the standalone security boundary.
// Builds dist-filler then asserts the public bundle has ZERO firebase/OPD code,
// and that the filler formula constants are obfuscated away (R9 parity).
// Exit 1 on any violation. Run manually + as the pre-deploy gate in deploy:filler.
import { execSync } from 'node:child_process';
import { readdirSync, readFileSync, statSync } from 'node:fs';
import { join } from 'node:path';

execSync('npx vite build --config vite.filler.config.js', { stdio: 'inherit' });

const walk = (d) => readdirSync(d).flatMap((f) => {
  const p = join(d, f);
  return statSync(p).isDirectory() ? walk(p) : [p];
});
const files = walk('dist-filler').filter((f) => f.endsWith('.js') || f.endsWith('.html'));
const blob = files.map((f) => readFileSync(f, 'utf8')).join('\n');

const FORBIDDEN = [
  /firebaseConfig/, /getFirestore/, /signInWith/, /firebase\/app/, /firebase\/firestore/,
  /AdminDashboard/, /BackendDashboard/, /backendClient/,
];
const FORMULA = [/K_REALISTIC/, /K_OPTIMISTIC/, /condomIndexForGirth/, /dCgeo/];

const hits = [];
for (const re of FORBIDDEN) if (re.test(blob)) hits.push('FORBIDDEN OPD/firebase token: ' + re);
for (const re of FORMULA) if (re.test(blob)) hits.push('Formula constant NOT obfuscated: ' + re);

if (hits.length) { console.error('❌ filler bundle FAILED:\n' + hits.join('\n')); process.exit(1); }
console.log(`✅ dist-filler verified: ${files.length} files, no firebase/OPD, formula obfuscated.`);
