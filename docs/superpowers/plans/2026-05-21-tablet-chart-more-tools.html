Tablet Chart Editor — Pro Toolset Implementation Plan

For agentic workers: Execute task-by-task (inline via executing-plans — chosen because the project baseline thrashes subagents, per V81/V86). Steps use HTML checkboxes for tracking. Spec: docs/superpowers/specs/2026-05-21-tablet-chart-more-tools-design.html.

Goal: Upgrade the tablet chart editor from a flat raster canvas to a full Fabric v7 object editor (select/move/resize + line/arrow/rect/circle/text + color picker), keeping the perfect-freehand pressure pen, and transporting a lossless fabricJson to the PC alongside the flattened PNG.

Architecture: Fabric v7 object model (reuse PC ChartCanvas patterns: locked template image, shapes, Textbox, JSON-history undo/redo, toDataURL). Pen = direct pointer-capture (mirror PenCanvas palm-rejection/coalesced) → fabric.Path from perfect-freehand outline on pointer-up (NOT BaseBrush subclass — avoids v7 brush-internal assumptions). Eraser = object-granular tap+scrub. Relay protocol unchanged; transport payload grows by one JSON Storage blob.

Tech Stack: React 19, Fabric v7.2 (lazy-imported), perfect-freehand 1.2, Firebase Storage, Vitest, Claude Preview (L1) + admin/client-SDK e2e (L2).

Verification mandate (user, this turn): EVERY tool must draw 100%, be erasable 100%, and round-trip to the PC carrying its object 100% — verified per Rule Q (real browser L1 + real-SDK L2), not mocks.

   resultImageUrl, resultFabricJsonUrl }                              ─▶ ChartSection.handleSave → charts[]  (PNG shows; fabricJson stored)
                                                                      ─▶ delete session + cleanup Storage + free tablet (existing teardown)
Tool interaction
 setTool(id) → TabletChartCanvas reacts:
   pen/highlighter → isDrawingMode off; OUR pointer capture: down→move(coalesced)→up
                     → fabric.Path(outlineToSvgPath(getStroke(pts))) added (selectable)
   line/arrow/rect/circle → isDrawingMode off; drag start→end → add fabric object → auto setTool('select')
   text            → tap → fabric.Textbox (editing) → type → commit
   select          → tap object → handles: drag move / corner resize / rotate
   eraser          → tap object → remove; drag → remove every object the path intersects
   undo/redo       → loadFromJSON(history[idx∓1])   ·   clear → remove all but locked template
 every commit → pushHistory() (toJSON snapshot, 40 deep)

// append to src/lib/penStroke.js
// outline = array of [x,y] (perfect-freehand getStroke output). Build an SVG path
// d-string (fabric.Path consumes a d-string, unlike Path2D). Numbers rounded to 2dp.
export function outlineToSvgPath(outline) {
  if (!outline || !outline.length) return '';
  const r = (n) => Math.round(n * 100) / 100;
  let d = `M${r(outline[0][0])} ${r(outline[0][1])}`;
  for (let i = 1; i < outline.length; i++) d += ` L${r(outline[i][0])} ${r(outline[i][1])}`;
  return d + ' Z';
}
// src/lib/tabletChartTools.js  (pure — no fabric/react import)
export const TOOL_IDS = ['select','pen','highlighter','line','arrow','rect','circle','text','eraser'];
export const isDrawTool  = (t) => t === 'pen' || t === 'highlighter';
export const isShapeTool = (t) => t === 'line' || t === 'arrow' || t === 'rect' || t === 'circle';
const TYPE = { pen:'path', highlighter:'path', line:'line', arrow:'group', rect:'rect', circle:'ellipse', text:'textbox' };
export function shapeObjectType(tool) { return TYPE[tool] || 'path'; }

// add to src/lib/chartEditSession.js (uses the existing `folder` + storage import)
export async function uploadTransportJson(sessionId, kind, obj) {
  const r = storageRef(storage, `${folder(sessionId)}/${kind}.json`);
  await uploadString(r, JSON.stringify(obj), 'raw', { contentType: 'application/json' });
  return getDownloadURL(r);
}
// Guarded: returns null on any failure so the PC merge never hangs (mirror result-download guard).
export async function downloadTransportJson(url) {
  try {
    const res = await fetch(url);
    if (!res.ok) return null;
    return JSON.parse(await res.text());
  } catch { return null; }
}

import { useRef, useEffect, useImperativeHandle, forwardRef, useCallback } from 'react';
import { strokeOutline, outlineToSvgPath, PEN_PRESETS } from '../../lib/penStroke.js';
import { isDrawTool, isShapeTool } from '../../lib/tabletChartTools.js';

// Fabric v7 object editor for the tablet. Same ref API as PenCanvas + extras.
// Pen/highlighter use OUR pointer capture (palm rejection + coalesced events, mirror PenCanvas)
// → fabric.Path from the perfect-freehand outline on pointerup (selectable object).
// Everything else (select/shapes/text/eraser) is Fabric-native.
const TabletChartCanvas = forwardRef(function TabletChartCanvas({ templateImageUrl, tool, color, size }, ref) {
  const elRef = useRef(null);          // 
  const fcRef = useRef(null);          // fabric.Canvas
  const fabricRef = useRef(null);      // fabric module
  const toolRef = useRef({ tool, color, size }); toolRef.current = { tool, color, size };
  const histRef = useRef([]); const hiRef = useRef(-1);
  const penRef = useRef(null);         // active freehand stroke { points, color, size, tool }
  const penDownRef = useRef(false);    // palm rejection: a pen pointer is active
  const shapeRef = useRef(null);       // in-progress shape { obj, startX, startY }

  const pushHistory = useCallback(() => {
    const fc = fcRef.current; if (!fc) return;
    const json = JSON.stringify(fc.toJSON());
    const h = histRef.current; h.length = hiRef.current + 1; h.push(json);
    if (h.length > 40) h.shift(); hiRef.current = h.length - 1;
  }, []);

  // init fabric + load locked template at true ratio (mirror ChartCanvas fit logic)
  useEffect(() => {
    let cancelled = false;
    (async () => {
      const fabric = await import('fabric'); fabricRef.current = fabric;
      if (cancelled || !elRef.current) return;
      const wrap = elRef.current.parentElement;
      const maxW = (wrap?.clientWidth || 700) - 8, maxH = (wrap?.clientHeight || 800) - 8;
      const mkImg = (src) => new Promise((res) => { const i = new window.Image(); i.crossOrigin = 'anonymous'; i.onload = () => res(i); i.onerror = () => res(null); i.src = src; });
      let cw = Math.min(maxW, Math.round(maxH * 0.78)), ch = maxH, imgEl = null;
      if (templateImageUrl) {
        imgEl = await mkImg(templateImageUrl);
        if (imgEl) { const ar = imgEl.naturalWidth / imgEl.naturalHeight;
          if (maxW / maxH > ar) { ch = maxH; cw = Math.round(maxH * ar); } else { cw = maxW; ch = Math.round(maxW / ar); } }
      }
      const fc = new fabric.Canvas(elRef.current, { width: cw, height: ch, backgroundColor: '#fff', isDrawingMode: false, selection: true, enableRetinaScaling: true });
      fcRef.current = fc;
      if (imgEl) { const fi = new fabric.FabricImage(imgEl); const s = Math.min(cw / fi.width, ch / fi.height);
        fi.set({ scaleX: s, scaleY: s, left: Math.round((cw - fi.width * s) / 2), top: Math.round((ch - fi.height * s) / 2), selectable: false, evented: false, hoverCursor: 'default', originX: 'left', originY: 'top' });
        fc.add(fi); fc.sendObjectToBack(fi); }
      fc.renderAll();
      histRef.current = [JSON.stringify(fc.toJSON())]; hiRef.current = 0;
      fc.on('object:modified', pushHistory);
      // pen pointer capture on the upper canvas
      const upper = fc.upperCanvasEl;
      const pt = (e) => fc.getScenePoint(e);
      const onDown = (e) => {
        const { tool, color, size } = toolRef.current;
        if (e.pointerType === 'touch' && penDownRef.current) return;            // palm rejection
        if (e.pointerType === 'pen') penDownRef.current = true;
        if (isDrawTool(tool)) { upper.setPointerCapture?.(e.pointerId); const p = pt(e); penRef.current = { tool, color, size, points: [[p.x, p.y, e.pressure || 0.5]] }; }
        else if (isShapeTool(tool)) { upper.setPointerCapture?.(e.pointerId); const p = pt(e); startShape(fabric, fc, tool, color, size, p); }
        else if (tool === 'eraser') { eraseAt(fc, pt(e)); }
        else if (tool === 'text') { addText(fabric, fc, color, size, pt(e)); }
      };
      const onMove = (e) => {
        if (penRef.current) { const evs = e.getCoalescedEvents ? e.getCoalescedEvents() : [e]; for (const ev of evs) { const p = pt(ev); penRef.current.points.push([p.x, p.y, ev.pressure || 0.5]); } previewPen(fc); }
        else if (shapeRef.current && toolRef.current.tool === 'eraser') { /* n/a */ }
        else if (shapeRef.current) { updateShape(fc, pt(e)); }
        else if (toolRef.current.tool === 'eraser' && (e.buttons & 1)) { eraseAt(fc, pt(e)); }   // scrub-delete
      };
      const onUp = (e) => {
        if (e.pointerType === 'pen') penDownRef.current = false;
        if (penRef.current) { commitPen(fabric, fc); }
        else if (shapeRef.current) { commitShape(fc); }
      };
      upper.addEventListener('pointerdown', onDown); upper.addEventListener('pointermove', onMove);
      upper.addEventListener('pointerup', onUp); upper.addEventListener('pointercancel', onUp);
      fc.__teardown = () => { upper.removeEventListener('pointerdown', onDown); upper.removeEventListener('pointermove', onMove); upper.removeEventListener('pointerup', onUp); upper.removeEventListener('pointercancel', onUp); };
    })();
    return () => { cancelled = true; const fc = fcRef.current; if (fc) { fc.__teardown?.(); fc.dispose(); fcRef.current = null; } };
  }, [templateImageUrl, pushHistory]);

  // ── pen ──
  const previewPen = (fc) => { const ctx = fc.contextTop; fc.clearContext(ctx); const s = penRef.current; const out = strokeOutline(s.points, s.tool === 'highlighter' ? PEN_PRESETS.highlighter(s.size) : PEN_PRESETS.pen(s.size)); if (!out.length) return; ctx.save(); ctx.globalAlpha = s.tool === 'highlighter' ? 0.4 : 1; ctx.fillStyle = s.color; const d = outlineToSvgPath(out); ctx.fill(new Path2D(d)); ctx.restore(); };
  const commitPen = (fabric, fc) => { const s = penRef.current; penRef.current = null; fc.clearContext(fc.contextTop); if (s.points.length < 2) return; const out = strokeOutline(s.points, s.tool === 'highlighter' ? PEN_PRESETS.highlighter(s.size) : PEN_PRESETS.pen(s.size)); const d = outlineToSvgPath(out); if (!d) return; const path = new fabric.Path(d, { fill: s.color, opacity: s.tool === 'highlighter' ? 0.4 : 1, stroke: null, selectable: true, objectCaching: false }); fc.add(path); fc.requestRenderAll(); pushHistory(); };

  // ── shapes ──
  const startShape = (fabric, fc, tool, color, size, p) => {
    let obj;
    if (tool === 'rect') obj = new fabric.Rect({ left: p.x, top: p.y, width: 1, height: 1, fill: 'transparent', stroke: color, strokeWidth: size });
    else if (tool === 'circle') obj = new fabric.Ellipse({ left: p.x, top: p.y, rx: 1, ry: 1, fill: 'transparent', stroke: color, strokeWidth: size });
    else if (tool === 'line') obj = new fabric.Line([p.x, p.y, p.x, p.y], { stroke: color, strokeWidth: size });
    else if (tool === 'arrow') { const line = new fabric.Line([p.x, p.y, p.x, p.y], { stroke: color, strokeWidth: size }); const head = new fabric.Triangle({ left: p.x, top: p.y, width: size * 4, height: size * 4, fill: color, originX: 'center', originY: 'center' }); obj = new fabric.Group([line, head], { subTargetCheck: false }); obj.__isArrow = true; }
    if (obj) { fc.add(obj); shapeRef.current = { obj, sx: p.x, sy: p.y, tool, color, size }; }
  };
  const updateShape = (fc, p) => { const s = shapeRef.current; if (!s) return; const o = s.obj;
    if (s.tool === 'rect') o.set({ width: Math.abs(p.x - s.sx), height: Math.abs(p.y - s.sy), left: Math.min(p.x, s.sx), top: Math.min(p.y, s.sy) });
    else if (s.tool === 'circle') o.set({ rx: Math.abs(p.x - s.sx) / 2, ry: Math.abs(p.y - s.sy) / 2, left: Math.min(p.x, s.sx), top: Math.min(p.y, s.sy) });
    else if (s.tool === 'line') o.set({ x2: p.x, y2: p.y });
    else if (s.tool === 'arrow') rebuildArrow(fc, s, p);
    fc.requestRenderAll(); };
  const rebuildArrow = (fc, s, p) => { fc.remove(s.obj); const fabric = fabricRef.current; const line = new fabric.Line([s.sx, s.sy, p.x, p.y], { stroke: s.color, strokeWidth: s.size }); const ang = Math.atan2(p.y - s.sy, p.x - s.sx) * 180 / Math.PI + 90; const head = new fabric.Triangle({ left: p.x, top: p.y, width: s.size * 4, height: s.size * 4, fill: s.color, originX: 'center', originY: 'center', angle: ang }); const g = new fabric.Group([line, head]); g.__isArrow = true; fc.add(g); s.obj = g; };
  const commitShape = (fc) => { const s = shapeRef.current; shapeRef.current = null; if (!s) return; const o = s.obj; const tiny = (o.width != null && o.width < 3 && o.height != null && o.height < 3); if (tiny) fc.remove(o); else { fc.setActiveObject(o); pushHistory(); } fc.requestRenderAll(); setToolToSelect(); };

  // ── text ──
  const addText = (fabric, fc, color, size, p) => { const tb = new fabric.Textbox('ข้อความ', { left: p.x, top: p.y, fontSize: Math.max(16, size * 4), fill: color, fontFamily: 'sans-serif', width: 120 }); fc.add(tb); fc.setActiveObject(tb); tb.enterEditing?.(); tb.selectAll?.(); fc.requestRenderAll(); pushHistory(); setToolToSelect(); };

  // ── eraser (object-granular) ──
  const eraseAt = (fc, p) => { const fabric = fabricRef.current; const target = fc.getObjects().filter(o => o.selectable !== false).reverse().find(o => o.containsPoint?.(new fabric.Point(p.x, p.y)) || (o.getBoundingRect && pointInRect(p, o.getBoundingRect()))); if (target) { fc.remove(target); fc.requestRenderAll(); pushHistory(); } };
  const pointInRect = (p, r) => p.x >= r.left && p.x <= r.left + r.width && p.y >= r.top && p.y <= r.top + r.height;

  const setToolToSelect = () => { /* notify parent via prop callback handled in page; here we just leave Fabric in select mode */ const fc = fcRef.current; if (fc) fc.isDrawingMode = false; };

  // react to tool/color/size changes
  useEffect(() => { const fc = fcRef.current; if (!fc) return; fc.isDrawingMode = false; fc.selection = (tool === 'select'); fc.skipTargetFind = (tool !== 'select' && tool !== 'eraser'); fc.defaultCursor = tool === 'eraser' ? 'crosshair' : 'default'; fc.forEachObject(o => { if (o.selectable !== false) { o.selectable = (tool === 'select'); o.evented = (tool === 'select' || tool === 'eraser'); } }); fc.requestRenderAll(); }, [tool, color, size]);

  useImperativeHandle(ref, () => ({
    exportDataUrl: () => { const fc = fcRef.current; fc.discardActiveObject(); fc.requestRenderAll(); return fc.toDataURL({ format: 'png', quality: 1, multiplier: 2 }); },
    exportFabricJson: () => { const fc = fcRef.current; return fc ? JSON.stringify(fc.toJSON()) : null; },
    undo: async () => { if (hiRef.current <= 0) return; hiRef.current--; await fcRef.current.loadFromJSON(JSON.parse(histRef.current[hiRef.current])); reLock(); fcRef.current.requestRenderAll(); },
    redo: async () => { if (hiRef.current >= histRef.current.length - 1) return; hiRef.current++; await fcRef.current.loadFromJSON(JSON.parse(histRef.current[hiRef.current])); reLock(); fcRef.current.requestRenderAll(); },
    clear: () => { const fc = fcRef.current; const objs = fc.getObjects(); for (let i = objs.length - 1; i >= 0; i--) if (objs[i].selectable !== false) fc.remove(objs[i]); fc.requestRenderAll(); pushHistory(); },
    deleteSelected: () => { const fc = fcRef.current; const a = fc.getActiveObjects?.() || []; a.forEach(o => o.selectable !== false && fc.remove(o)); fc.discardActiveObject(); fc.requestRenderAll(); if (a.length) pushHistory(); },
  }), [pushHistory]);
  const reLock = () => { const fc = fcRef.current; fc.getObjects().forEach((o, i) => { if (i === 0) o.set({ selectable: false, evented: false }); }); };

  return ();
});
