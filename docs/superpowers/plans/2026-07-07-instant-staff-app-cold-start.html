// src/lib/freshGate.js — customer-facing surfaces render SERVER-CONFIRMED data
// ONLY (fresh-always contract from Mobile-Load 2026-06-16 Q1, preserved after
// the 2026-07-07 staff-SWR reversal). Persistence makes every onSnapshot fire
// a cache snapshot first; this wrapper drops those so a customer never sees a
// stale course balance / appointment time. AV206 classifier locks consumers.
import { onSnapshot } from 'firebase/firestore';
export function onSnapshotFresh(refOrQuery, onData, onError) {
  return onSnapshot(refOrQuery, { includeMetadataChanges: true }, (snap) => {
    if (snap.metadata.fromCache) return; // stale-while-revalidate NOT for customers
    onData(snap);
  }, onError);
}

Step 2: implement swrRead.js
// src/lib/swrRead.js — stale-while-revalidate orchestrator for one-shot loads
// (staff surfaces only — customers use freshGate.js). Cache leg is best-effort
// and NEVER paints an empty result (avoids "ไม่มีรายการ" flash from a cold
// cache); the server leg is the source of truth and its errors propagate to
// the caller's existing resilient/error path.
export async function swrRun({ cacheLoad, serverLoad, apply }) {
  let paintedFromCache = false;
  try {
    const c = await cacheLoad();
    if (c && c.hasData) { apply(c.data, { fromCache: true }); paintedFromCache = true; }
  } catch { /* cold cache — silent */ }
  const fresh = await serverLoad();          // throws → caller's error path
  apply(fresh, { fromCache: false });
  return { paintedFromCache };
}

// src/hooks/useSwrLoad.js — generic SWR for a staff tab's mount load.
// loader(opts) MUST forward opts.source into the data-layer getters.
import { useState, useEffect, useRef, useCallback } from 'react';
import { swrRun } from '../lib/swrRead.js';
export function useSwrLoad(loader, deps) {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [syncing, setSyncing] = useState(false);
  const [error, setError] = useState(null);
  const [reloadKey, setReloadKey] = useState(0);
  const alive = useRef(true);
  useEffect(() => { alive.current = true; return () => { alive.current = false; }; }, []);
  useEffect(() => {
    let cancelled = false;
    setLoading(true); setError(null);
    swrRun({
      cacheLoad: async () => { const d = await loader({ source: 'cache' }); return { hasData: hasAny(d), data: d }; },
      serverLoad: () => loader({}),
      apply: (d, { fromCache }) => { if (cancelled || !alive.current) return; setData(d); setLoading(false); setSyncing(fromCache); },
    }).catch((e) => { if (!cancelled && alive.current) { setError(e); setLoading(false); setSyncing(false); } })
      .finally(() => { if (!cancelled && alive.current) setSyncing(false); });
    return () => { cancelled = true; };
  // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [...deps, reloadKey]);
  const reload = useCallback(() => setReloadKey(k => k + 1), []);
  return { data, loading, syncing, error, reload };
}
function hasAny(d) {
