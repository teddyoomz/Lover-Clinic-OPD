Task A1 — deriveClaimKey (pure) + DUPLICATE_IDENTITY error

NEW src/lib/customerIdentity.js. TDD: write tests/customer-identity-claim-key.test.js first (13-digit citizen w/ dashes/spaces → CITIZEN:digits; passport trim+UPPER → PASSPORT:X; both → citizen wins; both empty → null; citizen ≠13 digits → null + invalid flag; adversarial: Thai dashes, leading-zero reject, ≤30-char passport).

export class DuplicateIdentityError extends Error {
  constructor(existingCustomerId, claimKey) { super('DUPLICATE_IDENTITY'); this.name='DuplicateIdentityError';
    this.code='DUPLICATE_IDENTITY'; this.existingCustomerId=existingCustomerId; this.claimKey=claimKey; }
}
export function deriveClaimKey(citizen_id, passport_id) {
  const c = String(citizen_id||'').replace(/[-\s]/g,'');
  if (/^\d{13}$/.test(c)) return `CITIZEN:${c}`;
  const p = String(passport_id||'').trim().toUpperCase().replace(/\s+/g,'');
  if (p && p.length<=30 && /^[A-Z0-9]+$/.test(p)) return `PASSPORT:${p}`;
  return null;
}
