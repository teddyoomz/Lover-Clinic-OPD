Flow

Session state machine
PC pick template → "แท็บเล็ต" → pick ready tablet
   │ [TX: re-check target idle → presence→busy + create session 'requested' + upload template]
   ▼ tablet onSnapshot sees session for ITS deviceId
📱 editor pops INSTANTLY → 'active' (tablet heartbeat starts) → draw → Save → upload result → 'saved'
   ▼ PC onSnapshot 'saved' → download → charts[] merge (dataURL) → delete session + Storage → presence→idle ✅
CANCEL/TIMEOUT (any pre-'saved' state):
   tablet Cancel → 'cancelled'/tablet → PC ❌ failed
   PC Cancel     → 'cancelled'/pc     → tablet 🔌 → standby
   heartbeat>30s → 'cancelled'/timeout → both exit
   → every terminal: presence→idle + Storage cleaned
Sequence — who writes what
#ActorAction
1Tabletstandby: upsert presence {idle, branchId, name}; heartbeat ~10s
2PClist idle+fresh presences in branch; user picks
3PCTX: re-read target idle → busy + create session 'requested'; upload template; write templateImageUrl
4TabletonSnapshot 'requested' for my deviceId → download template → open editor → 'active' + heartbeat
5TabletSave: upload result PNG → write resultImageUrl + 'saved'
6PConSnapshot 'saved' → download → charts[] merge → delete session + Storage → presence idle
7Bothwatchdog: other heartbeat >30s stale → 'cancelled'/timeout → exit + cleanup

export const SESSION_STATUS = Object.freeze({ REQUESTED: 'requested', ACTIVE: 'active', SAVED: 'saved', CANCELLED: 'cancelled' });
export const CANCELLED_BY = Object.freeze({ PC: 'pc', TABLET: 'tablet', TIMEOUT: 'timeout' });
export const HEARTBEAT_INTERVAL_MS = 10000;
export const HEARTBEAT_STALE_MS = 30000;
export const SESSION_MAX_AGE_MS = 60 * 60 * 1000; // 1h orphan cap (cron sweep)

const TERMINAL = new Set([SESSION_STATUS.SAVED, SESSION_STATUS.CANCELLED]);
export function isTerminal(status) { return TERMINAL.has(status); }

// V81-fix1: tolerate every Firestore Timestamp serialization shape + JS primitives.
export function toMillis(ts) {
  if (ts == null) return 0;
  if (typeof ts === 'number') return ts;
  if (ts instanceof Date) return ts.getTime();
  if (typeof ts === 'string') { const n = Date.parse(ts); return Number.isNaN(n) ? 0 : n; }
  if (typeof ts.toMillis === 'function') return ts.toMillis();
  if (typeof ts.seconds === 'number') return ts.seconds * 1000 + Math.floor((ts.nanoseconds || 0) / 1e6);
  if (typeof ts._seconds === 'number') return ts._seconds * 1000 + Math.floor((ts._nanoseconds || 0) / 1e6);
  return 0;
}

export function isHeartbeatStale(lastMs, nowMs, staleMs = HEARTBEAT_STALE_MS) {
  return (nowMs - toMillis(lastMs)) > staleMs;
}
export function isPresenceReady(presence, nowMs) {
  if (!presence || presence.status !== 'idle') return false;
  return !isHeartbeatStale(presence.lastHeartbeatAt, nowMs);
}

const EDGES = {
  requested: new Set(['active', 'cancelled']),
  active: new Set(['saved', 'cancelled']),
  saved: new Set(), cancelled: new Set(),
};
export function canTransition(from, to) { return !!EDGES[from] && EDGES[from].has(to); }

export function buildPresenceUpsert({ deviceId, deviceName, branchId, uid, byName, status = 'idle', nowMs = Date.now() }) {
  return { deviceId, deviceName: deviceName || '', branchId: branchId || '', status, byUid: uid || '', byName: byName || '', lastHeartbeatAt: nowMs, updatedAt: nowMs };
}
export function buildSessionCreate({ sessionId, branchId, pcDeviceId, pcUid, tabletDeviceId, tabletName, template, patientLabel, nowMs = Date.now() }) {
  return {
    sessionId, branchId: branchId || '', pcDeviceId, pcUid: pcUid || '', tabletDeviceId, tabletName: tabletName || '',
    status: SESSION_STATUS.REQUESTED, cancelledBy: null,
    template: { id: template?.id || '', name: template?.name || '', category: template?.category || '' },
    patientLabel: patientLabel || '', templateImageUrl: null, resultImageUrl: null,
    pcHeartbeatAt: nowMs, tabletHeartbeatAt: null,
    createdAt: nowMs, updatedAt: nowMs, expiresAt: nowMs + SESSION_MAX_AGE_MS,
  };
}

export function listenToChartTabletPresenceByBranch({ branchId, allBranches = false } = {}, onChange, onError) {
  const effective = (typeof branchId === 'string' && branchId) ? branchId : (allBranches ? null : resolveSelectedBranchId());
  if (!effective && !allBranches) { onChange?.([]); return () => {}; }
  const base = colRef('be_chart_tablet_presence');
  const q = (!allBranches && effective) ? query(base, where('branchId', '==', String(effective))) : base;
  return onSnapshot(q, snap => onChange(snap.docs.map(d => ({ ...d.data(), id: d.id }))), err => onError?.(err)); // V38 spread order: id last
}
export async function upsertChartTabletPresence(deviceId, { deviceName, branchId, uid, byName, status }) {

  await setDoc(docRef('be_chart_tablet_presence', deviceId),
    { ...buildPresenceUpsert({ deviceId, deviceName, branchId: bid, uid, byName, status }), lastHeartbeatAt: serverTimestamp(), updatedAt: serverTimestamp() },
    { merge: true });
}

export function listenToRequestedSessionForTablet({ branchId, tabletDeviceId } = {}, onChange, onError) {
  const bid = (typeof branchId === 'string' && branchId) ? branchId : resolveSelectedBranchId();
  if (!bid || !tabletDeviceId) { onChange?.(null); return () => {}; }
  const q = query(colRef('be_chart_edit_sessions'), where('branchId', '==', bid), where('tabletDeviceId', '==', tabletDeviceId), where('status', '==', 'requested'));
  return onSnapshot(q, snap => onChange(snap.docs.length ? { ...snap.docs[0].data(), id: snap.docs[0].id } : null), err => onError?.(err));
}
// TX guard: only create when the target tablet is genuinely idle+fresh → prevents 2 PCs driving one tablet.
export async function createChartEditSession(payload) {
  const bid = (typeof payload.branchId === 'string' && payload.branchId) ? payload.branchId : resolveSelectedBranchId();
  const presRef = docRef('be_chart_tablet_presence', payload.tabletDeviceId);
  const sesRef = docRef('be_chart_edit_sessions', payload.sessionId);
  await runTransaction(db, async (tx) => {
    const pres = await tx.get(presRef);
    if (!pres.exists() || !isPresenceReady(pres.data(), Date.now())) { const e = new Error('TABLET_BUSY'); e.code = 'TABLET_BUSY'; throw e; }
    tx.set(presRef, { status: 'busy', updatedAt: serverTimestamp() }, { merge: true });
    tx.set(sesRef, { ...buildSessionCreate({ ...payload, branchId: bid }), createdAt: serverTimestamp(), updatedAt: serverTimestamp() });
  });
}

}

}
export async function deleteChartEditSession(sessionId) { await deleteDoc(docRef('be_chart_edit_sessions', sessionId)); }

// Build a Path2D from an outline polygon (for canvas fill).
export function outlineToPath2D(outline) {
  const p = new Path2D();
  if (!outline.length) return p;
  p.moveTo(outline[0][0], outline[0][1]);
  for (let i = 1; i < outline.length; i++) p.lineTo(outline[i][0], outline[i][1]);
  p.closePath();
  return p;
}

export function shouldReap(session, nowMs) {
  if (!session) return false;
  if (isTerminal(session.status)) return (nowMs - toMillis(session.updatedAt)) > SESSION_MAX_AGE_MS; // GC old terminals
  const pcStale = isHeartbeatStale(session.pcHeartbeatAt, nowMs);
  const tbStale = session.tabletHeartbeatAt != null && isHeartbeatStale(session.tabletHeartbeatAt, nowMs);
  return pcStale || tbStale || (nowMs - toMillis(session.createdAt)) > SESSION_MAX_AGE_MS;
}
