export const PI = Math.PI;
export const K_REALISTIC = 2.37, K_OPTIMISTIC = 3.32, CM_PER_INCH = 2.54;
export const RANGES = { lengthCm:[8,18], diameterCm:[2.2,4.1], cc:[1,50] };
export const CONDOM_LADDER = [
  {label:'Super snug 45',w:45},{label:'Close fit 49',w:49},{label:'Regular 52',w:52},
  {label:'Regular+ 54',w:54},{label:'Large 56',w:56},{label:'Large+ 58',w:58},
  {label:'XL 60',w:60},{label:'XXL 64',w:64} ];
const num = (x,d=0)=> Number.isFinite(x)? x : d;
export const widthFromGirth   = C => num(C)*5;
export const girthFromWidth   = W => num(W)/5;
export const diameterFromGirth= C => num(C)/PI;
export const girthFromDiameter= d => num(d)*PI;
export const girthToRadiusCm  = C => num(C)/(2*PI);
export const cmToInch = cm => num(cm)/CM_PER_INCH;
export const inchToCm = inch => num(inch)*CM_PER_INCH;
export function condomIndexForGirth(C){
  const req = num(C)*5; let bi=0;
  for(let i=1;i<CONDOM_LADDER.length;i++){
    const d=Math.abs(CONDOM_LADDER[i].w-req), bd=Math.abs(CONDOM_LADDER[bi].w-req);
    if(d<bd || (d===bd && CONDOM_LADDER[i].w>CONDOM_LADDER[bi].w)) bi=i; }
  return bi; }
export function condomForGirth(C){ const i=condomIndexForGirth(C); return {index:i, ...CONDOM_LADDER[i]}; }
const dCgeo = (C0,L,V)=> Math.sqrt(C0*C0 + 4*PI*num(V)/Math.max(num(L),0.1)) - C0;
export function estimate({ lengthCm, baseGirthCm, fillerCc }){
  const C0=num(baseGirthCm), L=Math.max(num(lengthCm),0.1), V=Math.max(num(fillerCc),0);
  const g=dCgeo(C0,L,V);
  const dLow=K_REALISTIC*g, dHigh=K_OPTIMISTIC*g;
  const c1Low=C0+dLow, c1High=C0+dHigh;
  const cd0=condomForGirth(C0), cdL=condomForGirth(c1Low), cdH=condomForGirth(c1High);
  return { c0:C0, d0:diameterFromGirth(C0), condom0:cd0,
    c1Low, c1High, d1Low:diameterFromGirth(c1Low), d1High:diameterFromGirth(c1High),
    deltaCLow:dLow, deltaCHigh:dHigh,
    condomLow:cdL, condomHigh:cdH,
    sizesUpLow:cdL.index-cd0.index, sizesUpHigh:cdH.index-cd0.index }; }
