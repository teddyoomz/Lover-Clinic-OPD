For agentic workers: REQUIRED SUB-SKILL: superpowers:executing-plans (inline, batch+checkpoint) or superpowers:subagent-driven-development. Steps use HTML checkboxes. Spec: docs/superpowers/specs/2026-06-02-scheduled-tasks-tab-design.html.

Goal: One admin-only backend tab ("งานอัตโนมัติ & ตารางเวลา") that consolidates all 10 scheduled tasks (after retiring a duplicate Firebase fn) — each enable/disable + tune params + see last-run status + run-now, at runtime.

Architecture: Extend the proven system_config doc with a scheduledTasks map; a pure scheduledTasksRegistry drives both the UI and the param defaults; each Vercel cron gets a 6-line fail-safe config guard + writes a last-run slice into a scheduled_task_status doc; a registry-driven ScheduledTasksTab reads config + status; a run-now endpoint invokes the cron handler with the server-side CRON_SECRET.

Tech Stack: React 19 + Vite, Firebase (client SDK in UI, admin SDK in crons), Vitest, existing systemConfigClient.js + SectionCard patterns.

Rules in force: C1 (extract SectionCard + reuse cores) · I (flow-simulate at end) · K (work-first for the cron sweep + UI, test bank consolidated) · N (targeted during, full suite at batch end) · Q (L2 real client-SDK + real cron run before "verified"; no deploy without explicit "deploy"). // api/_lib/scheduledTaskRuntime.js — admin-SDK helpers shared by all crons.
const APP_ID = 'loverclinic-opd-4c39b';
const PREFIX = `artifacts/${APP_ID}/public/data`;
const CONFIG_DOC = `${PREFIX}/clinic_settings/system_config`;
const STATUS_DOC = `${PREFIX}/clinic_settings/scheduled_task_status`;

// FAIL-SAFE: any error / missing → enabled:true + empty params (never silently
// stop a safety-critical cron because of a config-read error). Caller uses
// `cfg.params.X ?? CORE_DEFAULT`.
export async function readScheduledTaskConfig(db, taskId) {
  try {
    const snap = await db.doc(CONFIG_DOC).get();
    const all = snap.exists ? (snap.data()?.scheduledTasks || {}) : {};
    const t = all[taskId] || {};
    return {
      enabled: t.enabled !== false,
      params: (t.params && typeof t.params === 'object' && !Array.isArray(t.params)) ? t.params : {},
    };
  } catch (e) {
    console.warn(`[readScheduledTaskConfig] ${taskId} read failed; FAIL-SAFE enabled:`, e.message);
    return { enabled: true, params: {} };
  }
}

export async function writeScheduledTaskStatus(db, taskId, { ok = true, summary = '', error = '', skipped = false } = {}) {
  try {
    await db.doc(STATUS_DOC).set(
      { [taskId]: { lastRunAt: new Date().toISOString(), ok, summary: String(summary).slice(0,200), error: String(error).slice(0,300), skipped } },
      { merge: true },
    );
  } catch (e) {
    console.warn(`[writeScheduledTaskStatus] ${taskId} write failed (non-fatal):`, e.message);
  }
}

const CRON_MODULE = {

  wholeSystemBackup:'../cron/whole-system-backup-daily.js', chatHistoryRetention:'../cron/chat-history-retention-sweep.js',

};

export default async function handler(req, res) {
  const auth = await verifyAdminOrPermissionToken(req, res, 'scheduled_task_management');
  if (!auth) return; // helper already wrote 401/403
  if (req.method !== 'POST') return res.status(405).json({ error:'METHOD_NOT_ALLOWED' });
  const { taskId } = req.body || {};
  if (!getTask(taskId) || !CRON_MODULE[taskId]) return res.status(400).json({ error:'UNKNOWN_TASK', taskId });

  const mod = await import(CRON_MODULE[taskId]);
  // Synthetic request carrying the server-side CRON_SECRET + force=1 (run even if disabled).
  const fakeReq = { method:'GET', headers:{ authorization:`Bearer ${process.env.CRON_SECRET}` }, query:{ force:'1' }, body:{ force:true } };
  let payload = null, status = 200;
  const fakeRes = { status(c){ status=c; return this; }, json(o){ payload=o; return this; } };
  await mod.default(fakeReq, fakeRes);
  return res.status(status).json({ ranBy: auth.email || auth.uid, taskId, result: payload });
}

// src/hooks/useScheduledTaskStatus.js
import { useEffect, useState } from 'react';
import { doc, onSnapshot } from 'firebase/firestore';
import { db, appId } from '../firebase.js';
export function useScheduledTaskStatus() {
  const [status, setStatus] = useState({});
  useEffect(() => {
    const ref = doc(db, 'artifacts', appId, 'public', 'data', 'clinic_settings', 'scheduled_task_status');
    return onSnapshot(ref, (s) => setStatus(s.exists() ? (s.data() || {}) : {}), () => setStatus({}));
  }, []);
  return status;
}
