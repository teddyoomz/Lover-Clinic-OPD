// src/lib/staffChatOfficePreviewCore.js
//
// Pure JS — no React, no Firebase. Shared by client (UI state derivation),
// server (Cloud Function MIME gate), and Cloud Function (failure-reason copy).
// Rule of 3 alignment — single source of truth for "what counts as Office".
//
// Scope per spec Q3=C: Word + Excel + PowerPoint + CSV.
// Format: closed Set of canonical MIME strings (lowercase, no parameters).

export const OFFICE_CONVERTIBLE_MIMES = new Set([
  'application/msword',                                                                  // .doc
  'application/vnd.openxmlformats-officedocument.wordprocessingml.document',             // .docx
  'application/vnd.ms-excel',                                                            // .xls
  'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet',                   // .xlsx
  'application/vnd.ms-powerpoint',                                                       // .ppt
  'application/vnd.openxmlformats-officedocument.presentationml.presentation',           // .pptx
  'text/csv',                                                                            // .csv
]);

export function isOfficeConvertible(mime) {
  if (typeof mime !== 'string') return false;
  return OFFICE_CONVERTIBLE_MIMES.has(mime.toLowerCase().trim());
}

export const OfficePreviewStatus = Object.freeze({
  PENDING: 'pending',
  READY: 'ready',
  FAILED: 'failed',
  UNSUPPORTED: 'unsupported',
});

// Pure derivation — UI state from an attachment record. Returns one of:
//   'na'          — not an Office attachment; no preview affordance
//   'pending'     — Office MIME, conversion in flight (or status missing on legacy)
//   'ready'       — Office MIME, PDF cached, has pdfPreviewUrl
//   'failed'      — Office MIME, conversion failed
//   'unsupported' — Office MIME the server rejected (shouldn't happen for the 7 canonical MIMEs;
//                   reserved for future extension or rare LibreOffice format-not-supported errors)
export function pdfPreviewStateOf(att) {
  if (!att || typeof att !== 'object') return 'na';
  if (!isOfficeConvertible(att.mimeType)) return 'na';
  const status = att.pdfPreviewStatus;
  if (status === OfficePreviewStatus.READY) {
    // Defensive: ready requires a URL. Without one, fall back to pending.
    return (typeof att.pdfPreviewUrl === 'string' && att.pdfPreviewUrl.length > 0)
      ? OfficePreviewStatus.READY
      : OfficePreviewStatus.PENDING;
  }
  if (status === OfficePreviewStatus.FAILED) return OfficePreviewStatus.FAILED;
  if (status === OfficePreviewStatus.UNSUPPORTED) return OfficePreviewStatus.UNSUPPORTED;
  // Any other value (missing, undefined, garbage) → pending (legacy/inflight)
  return OfficePreviewStatus.PENDING;
}

  const m = typeof mime === 'string' ? mime.toLowerCase().trim() : '';
  if (STAFF_CHAT_RENDERABLE_IMAGE.has(m)) return 'image';

//
// (2026-05-22 EOD+2) Storage onObjectFinalized → LibreOffice (via bundled
// Gotenberg HTTP) → PDF back to Storage → Firestore patch. Fire-and-forget.
// All PHI stays inside our Firebase project (no 3rd-party HTTP).

import { onObjectFinalized } from 'firebase-functions/v2/storage';
import { initializeApp } from 'firebase-admin/app';
import { getStorage } from 'firebase-admin/storage';
import { getFirestore, FieldValue } from 'firebase-admin/firestore';
import { isOfficeConvertible, OfficePreviewStatus } from '../../src/lib/staffChatOfficePreviewCore.js';
import { deriveOutputPath, deriveFailureReason } from './helpers.js';

initializeApp();
const APP_ID = 'loverclinic-opd-4c39b';
const STAFF_CHAT_PREFIX = 'staff-chat-attachments/';
const GOTENBERG_URL = 'http://localhost:3000/forms/libreoffice/convert';

export const officeToPdf = onObjectFinalized(
  {
    region: 'asia-southeast1',
    memory: '1GiB',
    timeoutSeconds: 540,
    concurrency: 1,
  },
  async (event) => {
    const obj = event.data;
    const filePath = obj.name || '';
    const contentType = obj.contentType || '';

    // Gate 1: only files under the staff-chat prefix
    if (!filePath.startsWith(STAFF_CHAT_PREFIX)) return;
    // Gate 2: skip already-converted PDFs
    if (filePath.endsWith('.pdf')) return;
    // Gate 3: MIME whitelist (Q3=C scope)
    if (!isOfficeConvertible(contentType)) return;

    // The filePath under staff-chat is:
    //   staff-chat-attachments/{branchId}/{messageId}/{attachmentId}-o.{ext}
    // We need branchId + messageId + attachmentId to patch the right doc.
    const parts = filePath.slice(STAFF_CHAT_PREFIX.length).split('/');
    if (parts.length < 3) return;
    const [branchId, messageId, fileName] = parts;
    const attachmentId = (fileName.split('-o.')[0]) || fileName.split('.')[0];

    const bucket = getStorage().bucket();
    const db = getFirestore();
    const messageRef = db.collection('be_staff_chat_messages').doc(messageId);

    const stampStatus = async (patch) => {
      // Patch ONLY the matching attachments[i] entry (NOT the whole array — race-safe).
      // Read-then-write: get attachments, find by id, update field, write back.
      const snap = await messageRef.get();
      if (!snap.exists) return;
      const data = snap.data();
      const atts = Array.isArray(data.attachments) ? data.attachments.slice() : [];
      const idx = atts.findIndex(a => a && a.id === attachmentId);
      if (idx === -1) return;
      atts[idx] = { ...atts[idx], ...patch, pdfPreviewedAt: FieldValue.serverTimestamp() };
      await messageRef.update({ attachments: atts });
    };

    try {
      // 1. Download the original from Storage
      const file = bucket.file(filePath);
      const [buf] = await file.download();

      // 2. POST to bundled Gotenberg → PDF buffer
      const formData = new FormData();
      formData.append('files', new Blob([buf], { type: contentType }), fileName);
      const r = await fetch(GOTENBERG_URL, { method: 'POST', body: formData });
      if (!r.ok) {
        const txt = await r.text();
        let kind = 'unknown';
        if (/password|encrypt/i.test(txt)) kind = 'password-protected';
        else if (/corrupt|invalid|malformed/i.test(txt)) kind = 'corrupt';
        else if (/timeout/i.test(txt)) kind = 'timeout';
        await stampStatus({
          pdfPreviewStatus: OfficePreviewStatus.FAILED,
          pdfPreviewUrl: null,
          pdfPreviewError: deriveFailureReason({ kind }),
        });
        return;
      }
      const pdfBuf = Buffer.from(await r.arrayBuffer());

      // 3. Upload PDF to the same prefix
      const outPath = deriveOutputPath(filePath);
      const outFile = bucket.file(outPath);
      await outFile.save(pdfBuf, {
        contentType: 'application/pdf',
        metadata: { metadata: { generatedBy: 'officeToPdf', sourcePath: filePath } },
      });

      // 4. Make readable URL — use getDownloadURL pattern (long-lived token).
      //    (Per spec open question we choose getDownloadURL over 7d signed URLs to
      //     match V73 retention lifetime — PDF stays accessible until V73 sweep
      //     deletes both files together.)
      const downloadToken = (Math.random().toString(36).slice(2) + Math.random().toString(36).slice(2)).slice(0, 36);
      await outFile.setMetadata({ metadata: { firebaseStorageDownloadTokens: downloadToken } });
      const pdfUrl = `https://firebasestorage.googleapis.com/v0/b/${bucket.name}/o/${encodeURIComponent(outPath)}?alt=media&token=${downloadToken}`;

      // 5. Patch Firestore — status=ready + URL
      await stampStatus({
        pdfPreviewStatus: OfficePreviewStatus.READY,
        pdfPreviewUrl: pdfUrl,
        pdfPreviewError: null,
      });
    } catch (err) {
      console.error('[officeToPdf] conversion failed', { filePath, err: String(err) });
      await stampStatus({
        pdfPreviewStatus: OfficePreviewStatus.FAILED,
        pdfPreviewUrl: null,
        pdfPreviewError: deriveFailureReason({ kind: 'unknown' }),
      });
    }
  }
);

function walk(dir, out = []) {
  for (const e of readdirSync(dir)) {
    const full = join(dir, e);
    const s = statSync(full);
    if (s.isDirectory()) walk(full, out);
    else if (/\.(js|jsx|ts|tsx)$/.test(e)) out.push(full);
  }
  return out;
}
