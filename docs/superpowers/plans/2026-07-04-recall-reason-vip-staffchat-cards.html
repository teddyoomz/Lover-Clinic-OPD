// VIP context (2026-07-04) — ONE listener powers real-time VIP across every staff surface.
// Provider mounts ONLY in staff dashboards (BackendDashboard + AdminDashboard) —
// NEVER App root: anon users on public links would hit permission-denied.
// No provider => useIsVip returns false => VIP rendering silently off (structural
// guarantee for customer-facing pages).
import React, { createContext, useContext, useEffect, useState } from 'react';

const VipCtx = createContext(null);

export function VipProvider({ children }) {
  const [vipIds, setVipIds] = useState(() => new Set());
  useEffect(() => {
    let unsub = null; let cancelled = false;
    import('./scopedDataLayer.js').then(({ listenToVipCustomers }) => {
      if (cancelled) return;
      unsub = listenToVipCustomers(
        (ids) => setVipIds(new Set((ids || []).map(String))),
        () => setVipIds(new Set()),
      );
    });
    return () => { cancelled = true; unsub?.(); };
  }, []);
  return <VipCtx.Provider value={vipIds}>{children}</VipCtx.Provider>;
}

export function useIsVip(customerId) {
  const vipIds = useContext(VipCtx);
  if (!vipIds || !customerId) return false;
  return vipIds.has(String(customerId));
}

import { doc, setDoc, serverTimestamp } from 'firebase/firestore';
import { db } from '../firebase.js';

const APP_ID = 'loverclinic-opd-4c39b';
const COL = `artifacts/${APP_ID}/public/data/be_staff_chat_messages`;

export function buildTfpChatCard({ kind, treatmentId, customerId, customerName, customerHN, doctorName, branchId }) {
  const isDoctor = kind === 'tfp-doctor';
  return {
    id: `CHAT-SYS-TFP-${String(treatmentId)}-${isDoctor ? 'doctor' : 'vitals'}`,
    branchId: String(branchId || ''),
    deviceId: 'system',
    displayName: 'ระบบ',
    text: isDoctor ? '🔔 แพทย์ลงบันทึกเสร็จแล้ว' : '🔔 บันทึกซักประวัติเสร็จแล้ว',
    system: {
      kind: isDoctor ? 'tfp-doctor' : 'tfp-vitals',
      treatmentId: String(treatmentId || ''),
      customerId: String(customerId || ''),
      nameSnapshot: customerName || '',
      hnSnapshot: customerHN || null,
      ...(isDoctor ? { doctorName: doctorName || '' } : {}),
    },
  };
}

export async function writeTfpChatCard(payload) {
  try {
    const card = buildTfpChatCard(payload);
    if (!card.branchId || !card.system.treatmentId) return false;
    await setDoc(doc(db, `${COL}/${card.id}`), { ...card, createdAt: serverTimestamp() });
    return true;
  } catch (e) {
    console.warn('[tfp-chat-card] skipped:', e?.code || e?.message);  // rules-denied / duplicate / offline
    return false;
  }
}
