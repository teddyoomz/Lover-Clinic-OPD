/**
 * 2026-05-20 — validate a record-outcome submission.
 * Requires a non-empty outcome AND a recordedBy with a non-empty name (Q2=B).
 * @param {{outcome?:string, recordedBy?:{name?:string}}} payload
 * @returns {{ok:boolean, errors:string[]}}
 */
export function validateRecallOutcome(payload) {
  const errors = [];
  if (!payload?.outcome || typeof payload.outcome !== 'string') errors.push('outcome-required');
  const name = payload?.recordedBy?.name;
  if (typeof name !== 'string' || name.trim() === '') errors.push('recorded-by-required');
  return { ok: errors.length === 0, errors };
}
import { validateRecallOutcome } from '../src/lib/recallValidation.js';
test('ok with outcome + recordedBy.name', () => {
  expect(validateRecallOutcome({ outcome: 'will-come', recordedBy: { name: 'พิมพ์ชนก' } }).ok).toBe(true);
});
test('fails without staff', () => {
  const r = validateRecallOutcome({ outcome: 'will-come', recordedBy: { name: '' } });
  expect(r.ok).toBe(false); expect(r.errors).toContain('recorded-by-required');
});
test('fails without outcome', () => {
  expect(validateRecallOutcome({ recordedBy: { name: 'x' } }).errors).toContain('outcome-required');
});
