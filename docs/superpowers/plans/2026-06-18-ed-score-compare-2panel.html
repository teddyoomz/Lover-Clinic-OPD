// src/lib/edCompare.js — pure compare helpers for the ED 2-panel modal. No React.
export function autoPickCompareRound(rounds, primaryRound, type) {
  if (!Array.isArray(rounds) || !primaryRound) return null;
  const pd = primaryRound.assessmentDate || '';
  const others = rounds.filter(
    (r) => r && r.id !== primaryRound.id && Array.isArray(r.types) && r.types.includes(type)
  );
  if (!others.length) return null;
  const byDate = (a, b) => (a.assessmentDate || '').localeCompare(b.assessmentDate || '');
  const prior = others.filter((r) => (r.assessmentDate || '') < pd).sort(byDate);
  if (prior.length) return prior[prior.length - 1];     // closest before
  const later = others.filter((r) => (r.assessmentDate || '') >= pd).sort(byDate);
  return later.length ? later[0] : null;                // earliest after
}

export function markChangedRows(rowsPrimary, rowsCompare) {
  const a = Array.isArray(rowsPrimary) ? rowsPrimary : [];
  const b = Array.isArray(rowsCompare) ? rowsCompare : [];
  const map = (rows) => new Map(rows.map((r) => [r.n, r]));
  const other = (rows) => rows === a ? map(b) : map(a);
  const mark = (rows, o) => rows.map((r) => {
    const x = o.get(r.n);
    const changed = !!x && r.answer !== x.answer && r.answer !== '—' && x.answer !== '—';
    return { ...r, changed };
  });
  return { primary: mark(a, map(b)), compare: mark(b, map(a)) };
}

export default function EDDetailModal({ type, round, rounds = [], hero, isDark, onClose }) {
  const [activeType, setActiveType] = useState(type);
  const [primaryId, setPrimaryId] = useState(round?.id);
  const [compareId, setCompareId] =
    useState(() => autoPickCompareRound(rounds, round, type)?.id ?? null);

  useEffect(() => {  // ESC close (AV78 keeps backdrop non-closing)
    const onKey = (e) => { if (e.key === 'Escape') onClose?.(); };
    window.addEventListener('keydown', onKey);
    return () => window.removeEventListener('keydown', onKey);
  }, [onClose]);

  const { isPrimaryLeft, swap } = useLayoutPreference('ed-compare', 'left');

  const find = (id) => rounds.find((r) => r.id === id) || null;
  const primary = find(primaryId) || round;
  let compare = compareId ? find(compareId) : null;

  // keep compare meaningful when the type changes (re-pick if current compare lacks it)
  useEffect(() => {
    if (!compare || !compare.types?.includes(activeType)) {
      setCompareId(autoPickCompareRound(rounds, primary, activeType)?.id ?? null);
    }
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [activeType]);
  compare = compareId ? find(compareId) : null;

  if (!primary || !ED_TYPE_META[activeType]) return null;
  const tabs = ED_ORDER.filter((t) => rounds.some((r) => r.types?.includes(t)));
  const canCompare = !!compare;

  // rows + changed flags (only when both panels show the type)
  const pHas = primary.types?.includes(activeType);
  const cHas = compare?.types?.includes(activeType);
  let pRows = pHas ? buildEdAnswerRows(activeType, primary.raw) : null;
  let cRows = cHas ? buildEdAnswerRows(activeType, compare.raw) : null;
  if (pRows && cRows) { const m = markChangedRows(pRows, cRows); pRows = m.primary; cRows = m.compare; }

  const pPanel = <RoundPanel key="p" round={primary} hero={hero} isDark={isDark}
      activeType={activeType} rounds={rounds} rows={pRows}
      onPick={setPrimaryId} />;
  const cPanel = canCompare ? <RoundPanel key="c" round={compare} hero={hero} isDark={isDark}
      activeType={activeType} rounds={rounds} rows={cRows}
      onPick={setCompareId} muted /> : null;
  const order = isPrimaryLeft ? [pPanel, cPanel] : [cPanel, pPanel];

  return ( /* backdrop NO onClick (AV78); shell max-w-3xl if canCompare else max-w-xl;
             header: tabs (left) + swap btn (if canCompare) + ✕ (right);

                              : pPanel + hint "ยังไม่มีครั้งอื่นให้เทียบ" */ );
}
