Staff-Chat System Notification Cards — Implementation Plan

For agentic workers: implement task-by-task (executing-plans). Steps use HTML checkboxes. Spec: docs/superpowers/specs/2026-06-20-staff-chat-system-notification-cards-design.html.

Goal: When the intake-complete / follow-up-assessment-complete push fires, also write a "ระบบ" notification card into the per-branch staff chat — sparkles icon, customer name + HN, clickable name → customer detail in a new tab; intake cards live-upgrade to clickable+HN once the walk-in is registered.

Architecture: Server writes the card from the existing sendPushOnSubmit Cloud Function (admin SDK, after the FCM send, non-fatal). Client renders a dedicated StaffChatSystemCard with a render-time live-resolve hook (customerId direct for follow-up; via opd_session.brokerProClinicId for intake). No firestore.rules change.

Tech Stack: Firebase Cloud Functions (admin SDK, CJS), React + Vite, lucide-react, Firestore client SDK, Vitest + RTL, Playwright (L1), admin-SDK e2e (L2).

2pushFCM send (unchanged).
3write cardAfter FCM, non-fatal: kind=followup if linkedCustomerId else intake; SKIP if edit (updatedAt); branchId=session.branchId||customer.branchId; admin-SDK setDoc at be_staff_chat_messages.
4listenBranch listenToStaffChatMessages snapshot includes the card → V82 unread cursor + chime → counts unread + plays sound.
5renderStaffChatMessage: if (message.system) return <StaffChatSystemCard/>.
6resolveuseSystemCardCustomer: customerId from system.customerId (followup) OR live onSnapshot(opd_session) → brokerProClinicId (intake); then getCustomer → live name+HN.
7upgradeAdmin registers walk-in → handleOpdClick stamps brokerProClinicId → the session listener fires → the SAME card flips to clickable+HN, live, every viewer.

const crypto = require('crypto');
const { resolveCustomerName, resolveCustomerHN } = require('./customerDisplay');
function newId() { return `CHAT-${Date.now()}-${crypto.randomBytes(6).toString('hex')}`; }
function intakeName(session) {
  const pd = (session && session.patientData) || {};
  return `${pd.firstName || ''} ${pd.lastName || ''}`.trim();
}
function buildStaffChatNotification({ kind, session = {}, sessionId, customer, branchId, idFactory } = {}) {
  const isFollow = kind === 'followup';
  const name = isFollow ? (resolveCustomerName(customer) || String((session.confirmInfo && session.confirmInfo.name) || '').trim()) : intakeName(session);
  const customerId = isFollow ? String(session.linkedCustomerId || '') || null : null;
  const hn = isFollow ? (resolveCustomerHN(customer) || customerId || null) : null;
  const headline = isFollow ? 'กรอกแบบประเมินติดตามเสร็จแล้ว' : 'กรอกข้อมูลรับเข้าเสร็จแล้ว';
  return {
    id: (typeof idFactory === 'function' ? idFactory() : newId()),
    branchId: String(branchId || ''),
    deviceId: 'system',
    displayName: 'ระบบ',
    text: `🔔 ${headline}`,
    system: { kind: isFollow ? 'followup' : 'intake', sessionId: String(sessionId || ''), customerId: customerId || null, nameSnapshot: name || '', hnSnapshot: hn || null },
  };
}
async function writeStaffChatNotification(db, BASE_PATH, FieldValue, doc) {
  if (!doc || !doc.branchId) return false; // can't route without a branch — skip (non-fatal)
  await db.doc(`${BASE_PATH}/be_staff_chat_messages/${doc.id}`).set({ ...doc, createdAt: FieldValue.serverTimestamp() });
  return true;
}
module.exports = { buildStaffChatNotification, writeStaffChatNotification };

import { useEffect, useState } from 'react';
import { doc, onSnapshot } from 'firebase/firestore';
import { db, APP_ID } from '../firebase.js';
import { getCustomer } from './scopedDataLayer.js';
import { resolveCustomerDisplayName, resolveCustomerHN } from './customerDisplayName.js';

const BASE = `artifacts/${APP_ID}/public/data`;

export function pickSystemCardCustomerId(card, sessionData) {
  const sys = card && card.system; if (!sys) return null;
  if (sys.customerId) return String(sys.customerId);
  if (sessionData && sessionData.brokerProClinicId) return String(sessionData.brokerProClinicId);
  return null;
}

// Live: intake-unresolved subscribes to the opd_session so the card flips the instant it's registered.
export function useSystemCardCustomer(card) {
  const sys = (card && card.system) || {};
  const directId = sys.customerId ? String(sys.customerId) : '';
  const [customerId, setCustomerId] = useState(directId);
  const [resolved, setResolved] = useState(null); // { name, hn } from live be_customers

  // 1) intake → subscribe to the session until brokerProClinicId appears
  useEffect(() => {
    if (directId || !sys.sessionId) return;          // followup or no session → nothing to watch
    const ref = doc(db, BASE, 'opd_sessions', String(sys.sessionId));
    const unsub = onSnapshot(ref, (snap) => {
      const cid = snap.exists() ? snap.data().brokerProClinicId : '';
      if (cid) setCustomerId(String(cid));
    }, () => {});
    return () => unsub();
  }, [directId, sys.sessionId]);

  // 2) once a customerId is known → live name + HN
  useEffect(() => {
    if (!customerId) { setResolved(null); return; }
    let alive = true;
    getCustomer(customerId).then((c) => {
      if (alive && c) setResolved({ name: resolveCustomerDisplayName(c), hn: resolveCustomerHN(c) });
    }).catch(() => {});
    return () => { alive = false; };
  }, [customerId]);

  const pending = !customerId;
  return {
    pending,
    customerId,
    name: (resolved && resolved.name) || sys.nameSnapshot || '',
    hn: (resolved && resolved.hn) || sys.hnSnapshot || '',
  };
}
