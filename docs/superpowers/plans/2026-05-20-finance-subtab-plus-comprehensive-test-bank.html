A1: depositSubTabFilter helper + unit tests (TDD)

Create src/lib/depositSubTabFilter.js + tests/deposit-subtab-filter.test.js. Mirror saleSubTabFilter.

// src/lib/depositSubTabFilter.js
export const ACTIVE_DEPOSIT_STATUSES = ['active', 'partial'];
export const FINISHED_DEPOSIT_STATUSES = ['used', 'cancelled', 'refunded', 'expired'];

/** A deposit is "finished" iff its status is one of the terminal states.
 *  Mirrors the codebase's existing active|partial = usable convention
 *  (backendClient.js getDepositBalance / customer summary). */
export function isFinishedDeposit(dep) {
  return FINISHED_DEPOSIT_STATUSES.includes(dep?.status);
}

/** Partition loaded deposits for the given sub-tab.
 *  'finished' → terminal statuses; anything else → still-usable (active+partial).
 *  Defensive: non-array → []; null members tolerated. Unknown status (not in
 *  either set) defaults to the active pill (safer — stays visible/actionable). */
export function filterDepositsBySubTab(deposits, subTab) {
  const list = Array.isArray(deposits) ? deposits : [];
  if (subTab === 'finished') return list.filter(isFinishedDeposit);
  return list.filter((d) => !isFinishedDeposit(d));
}

Tests (groups A/B/C mirroring sale-subtab-filter): predicate (each of 6 statuses + null/missing), partition (active+partial→active pill; 4 finished→finished pill; complete+disjoint), adversarial (non-array, empty, null members, Thai name, unknown-status→active pill, idempotent, forward-compat). Run, expect PASS. Commit.
