For agentic workers: REQUIRED SUB-SKILL: superpowers:executing-plans (inline, recommended — tasks are tightly coupled + touch the large backendClient.js / AppointmentHubView.jsx baseline; subagents thrash here per V81/V86). Steps use HTML checkboxes for tracking.

Goal: Make the appointment-page customer card-list (OPD status stepper + appointment status + deposit/sale chips) update in REAL-TIME cross-device — any device that confirms/cancels/edits an appointment, saves an OPD/treatment, or changes a deposit/sale is reflected on every other device viewing the same page+branch, without a manual refresh, and stays live all day.

Architecture: Listener-trigger. Keep the existing one-shot loadAll. ADD live onSnapshot listeners for treatments + deposits + sales; each fires on a cross-device change → bumps a liveRefreshTick → the existing loadAll({silent:true}) re-fetches everything → derived state re-renders. Appointments are already live (AdminDashboard listenToAppointmentsByMonth, unchanged). Resilience via the existing V17 global reconnect (App.jsx visibilitychange/online → disable/enableNetwork) + a day-rollover/resume guard.

Tech Stack: React 19, Firebase Firestore onSnapshot, BSA layers (backendClient → scopedDataLayer → useBranchAwareListener), Vitest, Playwright (2-client L1).

// ── after listenToAllSales (~line 3497) ──
/**
 * Live trigger listener over be_treatments. Mirrors loadTreatmentsByDateRange
 * (reportsLoaders.js) — whole-collection read + client-filter (no server range
 * index on detail.treatmentDate). Used by the appointment page as a CHANGE
 * SIGNAL: any treatment create/edit/delete fires this → caller bumps a refresh
 * tick → loadAll re-fetches. allBranches:true mirrors the appointment page's
 * loadAll (V64-fix6 cross-branch auto-confirm). Returns unsubscribe.
 *
 * audit-branch-scope: report — supports {allBranches:true}; safe-by-default
 * (BS-13) for the branch-scoped path.
 */
export function listenToTreatmentsByDateRange(opts, onChange, onError) {
  const { from = '', to = '', includeCancelled = false } = opts || {};
  const allBranches = !!(opts && opts.allBranches);
  const explicitBranch = (opts && typeof opts.branchId === 'string' && opts.branchId) ? opts.branchId : '';
  const effectiveBranchId = explicitBranch || (allBranches ? '' : resolveSelectedBranchId());
  // Safe-by-default (BS-13): branch-scoped intent but no resolvable branch → no-op empty.
  if (!allBranches && !effectiveBranchId) { onChange?.([]); return () => {}; }
  return onSnapshot(treatmentsCol(), (snap) => {
    let items = snap.docs.map((d) => ({ ...d.data(), id: d.id }));
    if (from) items = items.filter((t) => (t?.detail?.treatmentDate || '') >= from);
    if (to) items = items.filter((t) => (t?.detail?.treatmentDate || '') <= to);
    if (!allBranches && effectiveBranchId) items = items.filter((t) => t.branchId === effectiveBranchId);
    if (!includeCancelled) items = items.filter((t) => t?.detail?.status !== 'cancelled');
    onChange?.(items);
  }, onError);
}

// ── after getAllDeposits (~line 4549) ──
/**
 * Live trigger listener over be_deposits. Mirrors getAllDeposits branch scope.
 * Used by the appointment page as a CHANGE SIGNAL. Returns unsubscribe.
 */
export function listenToAllDeposits(opts, onChange, onError) {
  const allBranches = !!(opts && opts.allBranches);
  const explicitBranch = (opts && typeof opts.branchId === 'string' && opts.branchId) ? opts.branchId : '';
  const effectiveBranchId = explicitBranch || (allBranches ? '' : resolveSelectedBranchId());
  if (!allBranches && !effectiveBranchId) { onChange?.([]); return () => {}; }
  const useFilter = !allBranches && effectiveBranchId;
  const q = useFilter
    ? query(depositsCol(), where('branchId', '==', String(effectiveBranchId)))
    : query(depositsCol());
  return onSnapshot(q, (snap) => {
    const list = snap.docs.map((d) => ({ ...d.data(), id: d.id }));
    list.sort((a, b) => (b.createdAt || '').localeCompare(a.createdAt || ''));
    onChange?.(list);
  }, onError);
}

import { listenToTreatmentsByDateRange, listenToAllDeposits, listenToAllSales } from '../../lib/scopedDataLayer.js';
import { thaiTodayISO } from '../../utils.js';
// (merge listenToAllSales into the existing scopedDataLayer import line if one exists)
// V-live (2026-05-27) — cross-device live: any listener fire bumps this tick;
// an effect re-runs loadAll({silent}). skip-first avoids the mount double-load.
const [liveRefreshTick, setLiveRefreshTick] = useState(0);
const liveFirstFire = useRef({ tx: true, dep: true, sale: true });
const [todayKey, setTodayKey] = useState(() => thaiTodayISO());
const wideRange = useMemo(() => {
  const past = dateRangeForTab('past', new Date());
  const future = dateRangeForTab('future', new Date());
  return { from: past.from, to: future.to };
// eslint-disable-next-line react-hooks/exhaustive-deps
}, [todayKey]);   // was [] — recompute on day-rollover
// Trigger: any live listener fire → silent reload. Coalesces bursts (React batches setState).
const liveRefreshTickPrev = useRef(0);
useEffect(() => {
  if (liveRefreshTick === liveRefreshTickPrev.current) return;
  liveRefreshTickPrev.current = liveRefreshTick;
  loadAll({ silent: true });
}, [liveRefreshTick, loadAll]);

const bumpLive = useCallback((key) => {
  if (liveFirstFire.current[key]) { liveFirstFire.current[key] = false; return; } // skip mount fire
  setLiveRefreshTick((t) => t + 1);
}, []);
const onTxLive = useCallback(() => bumpLive('tx'), [bumpLive]);
const onDepLive = useCallback(() => bumpLive('dep'), [bumpLive]);
const onSaleLive = useCallback(() => bumpLive('sale'), [bumpLive]);

// Treatments: allBranches:true (Q2 — mirror loadAll; V64-fix6). Direct useEffect.
// audit-branch-scope: listener-direct — allBranches trigger, BS-13 sanctioned exception.
useEffect(() => {
  const unsub = listenToTreatmentsByDateRange(
    { from: wideRange.from, to: wideRange.to, allBranches: true },
    onTxLive, () => {},
  );
  return () => { try { unsub?.(); } catch { /* defensive */ } };
}, [wideRange.from, wideRange.to, onTxLive]);

// Deposits + Sales: branch-scoped → useBranchAwareListener (auto branchId inject + re-subscribe on switch).
useBranchAwareListener(listenToAllDeposits, {}, onDepLive);
useBranchAwareListener(listenToAllSales, {}, onSaleLive);
// Resilience (req C) — long-open tab: refresh + roll the day window on resume.
// App.jsx V17 already resyncs onSnapshot listeners on visibility/online; this
// also rolls wideRange across midnight + forces a loadAll so a backgrounded tab
// is current the moment it returns.
useEffect(() => {
  const refresh = () => {
    if (document.visibilityState !== 'visible') return;
    const t = thaiTodayISO();
    setTodayKey((prev) => (prev !== t ? t : prev));
    setLiveRefreshTick((x) => x + 1);
  };
  document.addEventListener('visibilitychange', refresh);
  window.addEventListener('online', refresh);
  return () => {
    document.removeEventListener('visibilitychange', refresh);
    window.removeEventListener('online', refresh);
  };
}, []);

Acceptance Criteria

  1. Device A confirms/cancels/edits an appointment → Device B (same page+branch) reflects it with no manual refresh. (Already worked; must still pass.)
  2. Device A (doctor) saves OPD/treatment → Device B (admin) OPD stepper updates with no refresh. (The core gap being fixed.)
  3. Device A changes a deposit/sale → Device B card chips update with no refresh.
  4. A tab left open all day stays live; after background-suspend it catches up on resume (no manual refresh, listener does not die).
  5. No change to mutation logic, card rendering, OPD-stepper derivation, or business rules — purely added live subscriptions.
  6. Full suite green, build clean, /audit-branch-scope green.

Risks / Open Questions

  • Read cost: listenToTreatmentsByDateRange reads the whole be_treatments collection (mirrors the existing loadTreatmentsByDateRange one-shot — no new index, consistent). The listener only re-fires on the delta after the initial read, and each fire triggers one silent loadAll (the "reads มากขึ้นเล็กน้อย" trade the user accepted). If the collection grows very large, a future optimization is to bound the listener by a top-level createdAt window (single-field auto-index). Flagged, not implemented.
  • Redundant double-load on the saving device: the saving device gets both its existing same-session bump (treatmentRefreshKey / appointmentDataVersion) AND its own listener fire → up to 2 silent loadAlls. Harmless (idempotent silent re-fetch). Not deduped to keep the change minimal.
  • Branch-switch re-fire: useBranchAwareListener re-subscribes deposits/sales on branch switch; the new subscription's initial fire bumps the tick (skip-first is per-mount, not per-resubscribe) → one extra silent loadAll on switch. Harmless (loadAll already re-runs on branch switch via its own effect).
  • No commit/deploy without explicit authorization (V18). This plan implements + tests locally; deploy is a separate user-authorized step.
